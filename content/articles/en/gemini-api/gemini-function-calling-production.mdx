## Introduction: Function Calling + Structured Output Together

Gemini's **Function Calling** and **responseSchema (structured output)** are distinct features — but together they form the backbone of reliable, type-safe AI pipelines.

```
Function Calling:      LLM invokes external tools/APIs
       ↕
responseSchema:        LLM output is constrained to a JSON schema
```

This guide implements production-ready patterns in both Python and TypeScript.

<Callout type="info">
You'll need the Google AI SDK (`google-generativeai` for Python, `@google/generative-ai` for TypeScript). Set your `GOOGLE_API_KEY` as an environment variable before running any examples.
</Callout>

## 1. Basic Function Calling

### Python Implementation

```python
# function_calling_basic.py
import google.generativeai as genai
import json

genai.configure(api_key="YOUR_API_KEY")

WEATHER_TOOL = genai.protos.Tool(
    function_declarations=[
        genai.protos.FunctionDeclaration(
            name="get_weather",
            description="Get current weather for a specified city",
            parameters=genai.protos.Schema(
                type=genai.protos.Type.OBJECT,
                properties={
                    "city": genai.protos.Schema(
                        type=genai.protos.Type.STRING,
                        description="City name (e.g. Tokyo, New York)",
                    ),
                    "unit": genai.protos.Schema(
                        type=genai.protos.Type.STRING,
                        enum=["celsius", "fahrenheit"],
                    ),
                },
                required=["city"],
            ),
        )
    ]
)

def get_weather(city: str, unit: str = "celsius") -> dict:
    # In production: call Open-Meteo or OpenWeatherMap
    mock = {
        "Tokyo": {"temp": 22, "condition": "Clear", "humidity": 60},
        "New York": {"temp": 18, "condition": "Partly Cloudy", "humidity": 55},
    }
    data = mock.get(city, {"temp": 20, "condition": "Unknown", "humidity": 50})
    temp = data["temp"]
    if unit == "fahrenheit":
        temp = temp * 9 / 5 + 32
    return {**data, "temp": temp, "unit": unit, "city": city}

TOOL_REGISTRY = {"get_weather": get_weather}

def dispatch_tool_call(function_call) -> str:
    func = TOOL_REGISTRY.get(function_call.name)
    if not func:
        return json.dumps({"error": f"Unknown tool: {function_call.name}"})
    try:
        return json.dumps(func(**dict(function_call.args)))
    except Exception as e:
        return json.dumps({"error": str(e)})

def run_with_tools(user_message: str) -> str:
    model = genai.GenerativeModel(
        model_name="gemini-2.0-flash",
        tools=[WEATHER_TOOL],
    )
    chat = model.start_chat()
    response = chat.send_message(user_message)

    while True:
        parts = response.candidates[0].content.parts
        function_calls = [p for p in parts if hasattr(p, "function_call")]
        if not function_calls:
            break

        tool_responses = []
        for part in function_calls:
            result = dispatch_tool_call(part.function_call)
            tool_responses.append(
                genai.protos.Part(
                    function_response=genai.protos.FunctionResponse(
                        name=part.function_call.name,
                        response={"result": json.loads(result)},
                    )
                )
            )
        response = chat.send_message(tool_responses)

    return response.text

if __name__ == "__main__":
    print(run_with_tools("Compare the weather in Tokyo and New York today."))
```

## 2. Structured Output with responseSchema

```python
# structured_output.py
import google.generativeai as genai
import json
from typing import TypedDict

genai.configure(api_key="YOUR_API_KEY")

ARTICLE_SCHEMA = {
    "type": "object",
    "properties": {
        "title": {
            "type": "string",
            "description": "Article title (max 60 characters)"
        },
        "summary": {
            "type": "string",
            "description": "1-2 sentence summary"
        },
        "sentiment": {
            "type": "string",
            "enum": ["positive", "negative", "neutral"]
        },
        "topics": {
            "type": "array",
            "items": {"type": "string"},
            "description": "Topic tags (max 5)"
        },
        "difficulty": {
            "type": "integer",
            "minimum": 1,
            "maximum": 5,
            "description": "Technical difficulty (1=beginner, 5=expert)"
        }
    },
    "required": ["title", "summary", "sentiment", "topics", "difficulty"]
}

def analyze_article(content: str) -> dict:
    model = genai.GenerativeModel(
        model_name="gemini-2.0-flash",
        generation_config=genai.GenerationConfig(
            response_mime_type="application/json",
            response_schema=ARTICLE_SCHEMA,
        ),
    )
    response = model.generate_content(f"Analyze this article:\n\n{content}")
    # Response is guaranteed valid JSON matching the schema
    return json.loads(response.text)

class ArticleAnalysis(TypedDict):
    title: str
    summary: str
    sentiment: str
    topics: list[str]
    difficulty: int

def analyze_typed(content: str) -> ArticleAnalysis:
    return ArticleAnalysis(**analyze_article(content))
```

## 3. Parallel Function Calling

```python
# parallel_function_calling.py
import asyncio
import json
import google.generativeai as genai

class ParallelToolAgent:
    """Agent supporting parallel tool execution for reduced latency"""

    def __init__(self, tools: list, model_name: str = "gemini-2.0-flash"):
        self.model = genai.GenerativeModel(
            model_name=model_name,
            tools=tools,
            tool_config={"function_calling_config": {"mode": "AUTO"}},
        )
        self.tool_registry: dict = {}

    def register_tool(self, name: str, func):
        self.tool_registry[name] = func

    async def _call_tool_async(self, function_call) -> genai.protos.Part:
        func = self.tool_registry.get(function_call.name)
        if not func:
            result = {"error": f"Unknown tool: {function_call.name}"}
        else:
            try:
                args = dict(function_call.args)
                if asyncio.iscoroutinefunction(func):
                    result = await func(**args)
                else:
                    loop = asyncio.get_event_loop()
                    result = await loop.run_in_executor(None, lambda: func(**args))
            except Exception as e:
                result = {"error": str(e)}

        return genai.protos.Part(
            function_response=genai.protos.FunctionResponse(
                name=function_call.name,
                response={"result": result},
            )
        )

    async def run(self, user_message: str, max_iterations: int = 10) -> str:
        loop = asyncio.get_event_loop()
        chat = self.model.start_chat()
        response = await loop.run_in_executor(
            None, lambda: chat.send_message(user_message)
        )

        for _ in range(max_iterations):
            parts = response.candidates[0].content.parts
            calls = [p for p in parts if hasattr(p, "function_call") and p.function_call.name]
            if not calls:
                break

            # Execute ALL tool calls in parallel — key performance optimization
            results = await asyncio.gather(
                *[self._call_tool_async(p.function_call) for p in calls]
            )
            response = await loop.run_in_executor(
                None, lambda: chat.send_message(list(results))
            )

        return response.text


# Example: Stock + News + FX in parallel
async def get_stock(symbol: str) -> dict:
    await asyncio.sleep(0.5)  # Simulate API latency
    return {"symbol": symbol, "price": 875.20, "change": "+1.8%"}

async def get_news(topic: str) -> dict:
    await asyncio.sleep(0.3)
    return {"topic": topic, "headlines": [f"{topic} news 1", f"{topic} news 2"]}

async def get_fx(from_cur: str, to_cur: str) -> dict:
    await asyncio.sleep(0.2)
    return {"pair": f"{from_cur}/{to_cur}", "rate": 149.5}


async def main():
    TOOLS = [
        genai.protos.Tool(function_declarations=[
            genai.protos.FunctionDeclaration(
                name="get_stock",
                description="Get stock price",
                parameters=genai.protos.Schema(
                    type=genai.protos.Type.OBJECT,
                    properties={"symbol": genai.protos.Schema(type=genai.protos.Type.STRING)},
                    required=["symbol"],
                ),
            ),
            genai.protos.FunctionDeclaration(
                name="get_news",
                description="Get latest news",
                parameters=genai.protos.Schema(
                    type=genai.protos.Type.OBJECT,
                    properties={"topic": genai.protos.Schema(type=genai.protos.Type.STRING)},
                    required=["topic"],
                ),
            ),
            genai.protos.FunctionDeclaration(
                name="get_fx",
                description="Get exchange rate",
                parameters=genai.protos.Schema(
                    type=genai.protos.Type.OBJECT,
                    properties={
                        "from_cur": genai.protos.Schema(type=genai.protos.Type.STRING),
                        "to_cur": genai.protos.Schema(type=genai.protos.Type.STRING),
                    },
                    required=["from_cur", "to_cur"],
                ),
            ),
        ])
    ]

    agent = ParallelToolAgent(tools=TOOLS)
    agent.register_tool("get_stock", get_stock)
    agent.register_tool("get_news", get_news)
    agent.register_tool("get_fx", get_fx)

    result = await agent.run(
        "Get NVDA stock price, AI industry news, and USD/JPY rate simultaneously, then summarize."
    )
    print(result)

asyncio.run(main())
```

## 4. TypeScript — Next.js API Route

```typescript
// app/api/gemini-agent/route.ts
import { GoogleGenerativeAI, FunctionCallingMode } from '@google/generative-ai';
import { NextRequest, NextResponse } from 'next/server';

const genai = new GoogleGenerativeAI(process.env.GOOGLE_API_KEY!);

const TOOLS = [
  {
    functionDeclarations: [
      {
        name: 'search_products',
        description: 'Search product catalog',
        parameters: {
          type: 'object' as const,
          properties: {
            query: { type: 'string', description: 'Search query' },
            maxResults: { type: 'number', description: 'Max results (default 5)' },
          },
          required: ['query'],
        },
      },
      {
        name: 'get_product_details',
        description: 'Get product details by ID',
        parameters: {
          type: 'object' as const,
          properties: {
            productId: { type: 'string', description: 'Product ID' },
          },
          required: ['productId'],
        },
      },
    ],
  },
];

async function searchProducts(query: string, maxResults = 5) {
  return {
    results: Array.from({ length: maxResults }, (_, i) => ({
      id: `prod_${i + 1}`,
      name: `${query} Product ${i + 1}`,
      price: 29.99 + i * 10,
    })),
  };
}

async function getProductDetails(productId: string) {
  return { id: productId, name: `Product ${productId}`, price: 49.99, stock: 12 };
}

const DISPATCH: Record<string, (args: any) => Promise<object>> = {
  search_products: ({ query, maxResults }: any) => searchProducts(query, maxResults),
  get_product_details: ({ productId }: any) => getProductDetails(productId),
};

export async function POST(req: NextRequest) {
  const { message } = await req.json();

  const model = genai.getGenerativeModel({
    model: 'gemini-2.0-flash',
    tools: TOOLS,
    toolConfig: { functionCallingConfig: { mode: FunctionCallingMode.AUTO } },
  });

  const chat = model.startChat();
  let response = await chat.sendMessage(message);

  for (let i = 0; i < 10; i++) {
    const calls = response.response.functionCalls();
    if (!calls?.length) break;

    const results = await Promise.all(
      calls.map(async (fc) => {
        const fn = DISPATCH[fc.name];
        const result = fn ? await fn(fc.args as any) : { error: `Unknown tool: ${fc.name}` };
        return { functionResponse: { name: fc.name, response: result } };
      }),
    );

    response = await chat.sendMessage(results);
  }

  return NextResponse.json({ result: response.response.text() });
}
```

## Summary

Gemini Function Calling + responseSchema gives you **type-safe, production-grade AI API pipelines** with predictable outputs and efficient parallelism.

Key takeaways: use `asyncio.gather` for parallel tool calls to eliminate sequential latency; use `responseSchema` to eliminate fragile JSON parsing; choose `gemini-2.0-flash` for most API workloads and reserve `gemini-2.5-pro` for tasks requiring deep reasoning.

Next step: explore Gemini Context Caching to reduce costs when processing the same large documents repeatedly.
