## Getting Started with Image Generation

The Gemini Image Generation API transforms text descriptions into high-quality visual content. Powered by Google's Imagen technology, it enables developers to create:

- Marketing visuals and social media content
- Concept art and design exploration
- Product mockups and prototypes
- Educational diagrams and illustrations
- Game assets and character designs
- Architectural visualizations

This guide covers everything from basic setup through advanced customization techniques.

## Setting Up Authentication

### Initial Configuration

```python
import google.generativeai as genai
import os
from pathlib import Path
from datetime import datetime

# Configure API access
api_key = os.getenv("GEMINI_API_KEY")
genai.configure(api_key=api_key)

def initialize_image_generation():
    """Initialize the image generation API"""
    # List available models
    models = genai.list_models()

    # Find image generation models
    image_models = [
        m for m in models
        if 'imagen' in m.name.lower() or 'image' in m.name.lower()
    ]

    if image_models:
        selected = image_models[0].name
        print(f"✓ Using model: {selected}")
        return selected
    else:
        # Fallback to default
        return "gemini-2.0-flash"

model = initialize_image_generation()
print(f"Ready to generate images with: {model}")

# Expected output:
# ✓ Using model: gemini-2.0-flash
# Ready to generate images with: gemini-2.0-flash
```

## Basic Image Generation

### Creating Images from Text

```python
def generate_simple_image(prompt: str, output_file: str = None) -> dict:
    """Generate a single image from text prompt"""
    client = genai.Client(api_key=os.getenv("GEMINI_API_KEY"))

    try:
        response = client.models.generate_images(
            model="gemini-2.0-flash",
            prompt=prompt,
            number_of_images=1,
            size_width=1024,
            size_height=1024,
            safety_filter_level="block_medium_and_above"
        )

        if response.images:
            image_data = response.images[0].data

            # Save to file
            if not output_file:
                output_file = f"image_{datetime.now().strftime('%Y%m%d_%H%M%S')}.png"

            with open(output_file, "wb") as f:
                f.write(image_data)

            return {
                "status": "success",
                "file": output_file,
                "size_bytes": len(image_data),
                "prompt": prompt
            }
        else:
            return {
                "status": "no_image",
                "reason": "API returned no images"
            }

    except Exception as e:
        return {
            "status": "error",
            "error_message": str(e)
        }

# Example: Generate a landscape
result = generate_simple_image(
    prompt="A misty mountain valley at sunrise with ancient temples",
    output_file="landscape.png"
)

print(f"Status: {result['status']}")
if result['status'] == 'success':
    print(f"Saved to: {result['file']}")

# Expected output:
# Status: success
# Saved to: landscape.png
```

## Prompt Engineering Techniques

### Crafting Effective Prompts

Image quality depends heavily on prompt quality. Specific, detailed descriptions yield better results than vague ones.

```python
def build_detailed_prompt(
    primary_subject: str,
    visual_style: str,
    environment: str,
    lighting: str,
    mood: str,
    art_direction: str = None
) -> str:
    """Construct a detailed, structured prompt"""
    components = [
        f"Main subject: {primary_subject}",
        f"Style: {visual_style}",
        f"Setting: {environment}",
        f"Lighting: {lighting}",
        f"Mood: {mood}"
    ]

    if art_direction:
        components.append(f"Art direction: {art_direction}")

    return ", ".join(components)

def generate_image_series(image_concepts: list) -> list:
    """Generate multiple images with different concepts"""
    client = genai.Client(api_key=os.getenv("GEMINI_API_KEY"))
    results = []

    for idx, concept in enumerate(image_concepts):
        # Build detailed prompt
        prompt = build_detailed_prompt(
            primary_subject=concept.get("subject"),
            visual_style=concept.get("style"),
            environment=concept.get("environment"),
            lighting=concept.get("lighting"),
            mood=concept.get("mood"),
            art_direction=concept.get("art_direction")
        )

        try:
            response = client.models.generate_images(
                model="gemini-2.0-flash",
                prompt=prompt,
                number_of_images=1,
                size_width=1024,
                size_height=1024
            )

            if response.images:
                filename = f"generated_{idx:03d}.png"
                with open(filename, "wb") as f:
                    f.write(response.images[0].data)

                results.append({
                    "index": idx,
                    "filename": filename,
                    "prompt": prompt,
                    "status": "generated"
                })
                print(f"✓ Generated: {filename}")
            else:
                results.append({
                    "index": idx,
                    "status": "no_image",
                    "prompt": prompt
                })

        except Exception as e:
            results.append({
                "index": idx,
                "status": "error",
                "error": str(e)
            })
            print(f"✗ Error generating image {idx}: {e}")

    return results

# Define image concepts
concepts = [
    {
        "subject": "Mechanical dragon",
        "style": "Steampunk illustration",
        "environment": "Industrial Victorian city",
        "lighting": "Golden hour sunlight through steam",
        "mood": "Majestic and mysterious",
        "art_direction": "High detail, metallic surfaces, gears visible"
    },
    {
        "subject": "Bioluminescent forest",
        "style": "Digital art",
        "environment": "Dense jungle at night",
        "lighting": "Glowing plants and creatures",
        "mood": "Magical and ethereal",
        "art_direction": "Neon colors, depth of field"
    }
]

results = generate_image_series(concepts)
successful = [r for r in results if r["status"] == "generated"]
print(f"\nGenerated {len(successful)} images successfully")

# Expected output:
# ✓ Generated: generated_000.png
# ✓ Generated: generated_001.png
# Generated 2 images successfully
```

## Advanced Customization

### Variable Resolutions and Aspect Ratios

```python
def generate_with_custom_resolution(
    prompt: str,
    width: int = 1024,
    height: int = 1024,
    count: int = 1
) -> dict:
    """Generate images with custom dimensions"""
    client = genai.Client(api_key=os.getenv("GEMINI_API_KEY"))

    # Supported resolutions by the API
    supported_resolutions = [
        (1024, 1024),  # Square
        (1344, 768),   # Landscape
        (768, 1024),   # Portrait
        (512, 512),    # Small square
        (1024, 768),   # Slightly landscape
    ]

    # Validate resolution
    if (width, height) not in supported_resolutions:
        print(f"⚠ {width}x{height} not supported. Using 1024x1024")
        width, height = 1024, 1024

    try:
        response = client.models.generate_images(
            model="gemini-2.0-flash",
            prompt=prompt,
            number_of_images=count,
            size_width=width,
            size_height=height
        )

        filenames = []
        for i, image in enumerate(response.images):
            filename = f"output_{width}x{height}_{i:02d}.png"
            with open(filename, "wb") as f:
                f.write(image.data)
            filenames.append(filename)

        return {
            "status": "success",
            "resolution": f"{width}x{height}",
            "count": len(filenames),
            "files": filenames
        }

    except Exception as e:
        return {
            "status": "error",
            "error": str(e)
        }

# Generate for social media (1344x768 landscape format)
result = generate_with_custom_resolution(
    prompt="Serene beach sunset with palm trees and calm waves",
    width=1344,
    height=768,
    count=2
)

print(f"Resolution: {result['resolution']}")
print(f"Generated: {result['count']} images")

# Expected output:
# Resolution: 1344x768
# Generated: 2 images
```

## Safety and Content Guidelines

### Managing Safety Filters

```python
def generate_with_safety_config(
    prompt: str,
    safety_level: str = "block_medium_and_above",
    allow_adult_content: bool = False
) -> dict:
    """Generate with explicit safety controls"""
    client = genai.Client(api_key=os.getenv("GEMINI_API_KEY"))

    safety_levels = {
        "block_none": "No filtering",
        "block_low_and_above": "Block low-risk and above",
        "block_medium_and_above": "Block medium-risk and above (default)",
        "block_high_and_above": "Block only high-risk content"
    }

    try:
        response = client.models.generate_images(
            model="gemini-2.0-flash",
            prompt=prompt,
            number_of_images=1,
            size_width=1024,
            size_height=1024,
            safety_filter_level=safety_level,
            person_generation_config=genai.types.PersonGenerationConfig(
                allow_adult=allow_adult_content
            )
        )

        return {
            "safety_level": safety_levels.get(safety_level),
            "generated": bool(response.images),
            "image_count": len(response.images) if response.images else 0
        }

    except Exception as e:
        return {
            "status": "error",
            "error": str(e)
        }

# Generate with safety filtering
result = generate_with_safety_config(
    prompt="Professional business meeting in a modern office",
    safety_level="block_medium_and_above",
    allow_adult_content=False
)

print(f"Safety: {result['safety_level']}")
print(f"Generated: {result['generated']}")
```

## Frequently Asked Questions

### Q1: Who owns the copyright to generated images?

**A:** You retain the copyright to images generated by the Gemini API. You can use them commercially, but review the terms of service. You're responsible for ensuring generated images don't infringe on existing copyrights or trademarks. Avoid using prompts that ask for exact replicas of copyrighted works.

### Q2: Can I generate images of real people?

**A:** No. The API's safety filters prevent generating photorealistic images of real, identifiable people. You can generate fictional characters or stylized representations, but not recreations of specific individuals.

### Q3: What should I do if image generation fails or is blocked?

**A:** Try simplifying your prompt, using different wording, or breaking down complex concepts. If filtering blocks content, you may be requesting something against the policy. Check your safety level settings and consider whether the prompt violates content guidelines.
