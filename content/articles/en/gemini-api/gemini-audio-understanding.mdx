## Understanding Audio Processing with Gemini

The Gemini Audio Understanding API converts speech to text and extracts meaningful insights from audio content. Use cases include:

- Podcast and webinar transcription
- Meeting recordings and summaries
- Content search and indexing
- Language learning support
- Customer service call analysis
- Automated quality assurance

This guide walks through everything from basic transcription to sophisticated analysis pipelines.

## Getting Started with Audio API

### Initial Setup

```python
import google.generativeai as genai
import os
import mimetypes
from pathlib import Path
from datetime import datetime

# Configure the API
api_key = os.getenv("GEMINI_API_KEY")

def initialize_audio_api():
    """Initialize audio processing"""
```

## Basic Audio Transcription

### Converting Speech to Text

```python
def transcribe_audio_file(audio_path: str) -> dict:
    """Transcribe an audio file to text"""
    client = genai.Client(api_key=os.getenv("GEMINI_API_KEY"))

    # Validate file exists
    if not Path(audio_path).exists():
        return {
            "status": "error",
            "message": f"File not found: {audio_path}"
        }

    # Determine MIME type
    mime_type, _ = mimetypes.guess_type(audio_path)

    # Fallback to extension-based detection
    if not mime_type or 'audio' not in mime_type:
        ext_to_mime = {
            '.mp3': 'audio/mpeg',
            '.wav': 'audio/wav',
            '.m4a': 'audio/mp4',
            '.flac': 'audio/flac',
            '.ogg': 'audio/ogg',
            '.webm': 'audio/webm'
        }
        ext = Path(audio_path).suffix.lower()
        mime_type = ext_to_mime.get(ext, 'audio/mpeg')

    # Read audio file
    try:
        with open(audio_path, 'rb') as f:
            audio_data = f.read()

        response = client.models.generate_content(
            model="gemini-2.0-flash",
            contents=[
                "Transcribe this audio file completely and accurately.",
                {
                    "mime_type": mime_type,
                    "data": audio_data
                }
            ]
        )

        transcript = response.text

        return {
            "status": "success",
            "file": audio_path,
            "transcript": transcript,
            "word_count": len(transcript.split()),
            "character_count": len(transcript)
        }

    except Exception as e:
        return {
            "status": "error",
            "message": str(e)
        }

# Example: Transcribe a meeting recording
result = transcribe_audio_file("meeting_recording.mp3")

if result["status"] == "success":
    print(f"✓ Transcription complete: {result['word_count']} words")
    print(f"First 200 characters:\n{result['transcript'][:200]}...")

# Expected output:
# ✓ Transcription complete: 1250 words
# First 200 characters:
# Good morning, everyone. Thank you for joining today's meeting...
```

## Advanced Audio Analysis

### Analyzing Podcast Content

```python
def analyze_audio_content(
    audio_path: str,
    analysis_type: str = "summary"
) -> dict:
    """Analyze audio with different focus areas"""
    client = genai.Client(api_key=os.getenv("GEMINI_API_KEY"))

    analysis_templates = {
        "summary": "Provide a concise summary highlighting the main points and key takeaways.",
        "topics": "List all major topics and themes discussed in this audio.",
        "sentiment": "Analyze the overall tone and emotional quality of the audio.",
        "keywords": "Extract the most important keywords and technical terms.",
        "speakers": "Identify different speakers if present, and summarize their contributions.",
        "action_items": "List any action items, decisions, or next steps mentioned.",
        "detailed": """Provide detailed analysis with these sections:
1. Executive Summary
2. Main Topics Discussed
3. Key Insights and Findings
4. Action Items and Decisions
5. Sentiment and Tone Analysis
6. Important Technical Terms
7. Recommendations for Follow-up"""
    }

    prompt = analysis_templates.get(analysis_type, analysis_templates["summary"])

    # Determine MIME type
    ext = Path(audio_path).suffix.lower()
    mime_map = {
        '.mp3': 'audio/mpeg',
        '.wav': 'audio/wav',
        '.m4a': 'audio/mp4',
        '.flac': 'audio/flac',
    }
    mime_type = mime_map.get(ext, 'audio/mpeg')

    try:
        with open(audio_path, 'rb') as f:
            audio_data = f.read()

        response = client.models.generate_content(
            model="gemini-2.0-flash",
            contents=[
                prompt,
                {
                    "mime_type": mime_type,
                    "data": audio_data
                }
            ]
        )

        return {
            "status": "success",
            "analysis_type": analysis_type,
            "file": Path(audio_path).name,
            "analysis": response.text
        }

    except Exception as e:
        return {
            "status": "error",
            "message": str(e)
        }

# Example: Analyze a podcast episode
podcast = "episode_42.mp3"

# Get summary
summary = analyze_audio_content(podcast, analysis_type="summary")
print("SUMMARY:")
print(summary["analysis"])

# Extract action items
actions = analyze_audio_content(podcast, analysis_type="action_items")
print("\nACTION ITEMS:")
print(actions["analysis"])

# Expected output:
# SUMMARY:
# This episode discusses recent developments in machine learning...
#
# ACTION ITEMS:
```

## Batch Processing Audio Files

### Processing Multiple Files Efficiently

```python
def process_audio_batch(audio_files: list, save_transcripts: bool = True) -> dict:
    """Process multiple audio files in sequence"""
    client = genai.Client(api_key=os.getenv("GEMINI_API_KEY"))

    results = {
        "total": len(audio_files),
        "successful": 0,
        "failed": 0,
        "details": []
    }

    for idx, audio_file in enumerate(audio_files):
        print(f"[{idx + 1}/{len(audio_files)}] Processing: {audio_file}")

        # Validate file
        if not Path(audio_file).exists():
            results["details"].append({
                "file": audio_file,
                "status": "error",
                "reason": "File not found"
            })
            results["failed"] += 1
            continue

        # Get MIME type
        ext = Path(audio_file).suffix.lower()
        mime_type = {
            '.mp3': 'audio/mpeg',
            '.wav': 'audio/wav',
            '.m4a': 'audio/mp4',
            '.flac': 'audio/flac',
        }.get(ext, 'audio/mpeg')

        try:
            with open(audio_file, 'rb') as f:
                audio_data = f.read()

            response = client.models.generate_content(
                model="gemini-2.0-flash",
                contents=[
                    "Transcribe this audio completely.",
                    {
                        "mime_type": mime_type,
                        "data": audio_data
                    }
                ]
            )

            transcript = response.text

            # Save transcript if requested
            if save_transcripts:
                output_file = Path(audio_file).stem + "_transcript.txt"
                with open(output_file, 'w', encoding='utf-8') as f:
                    f.write(transcript)

                print(f"✓ Saved: {output_file}")

            results["details"].append({
                "file": Path(audio_file).name,
                "status": "success",
                "output": output_file if save_transcripts else None,
                "length": len(transcript)
            })
            results["successful"] += 1

        except Exception as e:
            results["details"].append({
                "file": audio_file,
                "status": "error",
                "reason": str(e)
            })
            results["failed"] += 1
            print(f"✗ Error: {str(e)}")

    return results

# Example: Process multiple recordings
files = [
    "meeting_2026_01_15.mp3",
    "meeting_2026_01_22.mp3",
    "meeting_2026_01_29.mp3"
]

batch_results = process_audio_batch(files)
print(f"\nCompleted: {batch_results['successful']}/{batch_results['total']} successful")

# Expected output:
# [1/3] Processing: meeting_2026_01_15.mp3
# ✓ Saved: meeting_2026_01_15_transcript.txt
# [2/3] Processing: meeting_2026_01_22.mp3
# ✓ Saved: meeting_2026_01_22_transcript.txt
# [3/3] Processing: meeting_2026_01_29.mp3
# ✓ Saved: meeting_2026_01_29_transcript.txt
#
# Completed: 3/3 successful
```

## Comprehensive Audio Analytics

### Multi-Dimensional Analysis Pipeline

```python
def comprehensive_audio_analysis(audio_path: str) -> dict:
    """Perform comprehensive analysis across multiple dimensions"""
    client = genai.Client(api_key=os.getenv("GEMINI_API_KEY"))

    with open(audio_path, 'rb') as f:
        audio_data = f.read()

    # Complex analysis prompt
    analysis_prompt = """
    Analyze this audio comprehensively and return structured JSON with:

    {
      "transcription": "complete transcription",
      "summary": "brief executive summary",
      "speakers": ["speaker1", "speaker2"],
      "language": "detected language",
      "duration_estimate": "estimated duration",
      "main_topics": ["topic1", "topic2"],
      "sentiment": {
        "overall": "positive/neutral/negative",
        "confidence": 0.0
      },
      "key_quotes": ["important quote 1", "important quote 2"],
      "action_items": ["action1", "action2"],
      "keywords": ["keyword1", "keyword2"],
      "quality_assessment": "assessment of audio quality"
    }
    """

    mime_type = 'audio/mpeg'

    try:
        response = client.models.generate_content(
            model="gemini-2.0-flash",
            contents=[
                analysis_prompt,
                {
                    "mime_type": mime_type,
                    "data": audio_data
                }
            ]
        )

        import json

        # Try to parse as JSON
        try:
            analysis_data = json.loads(response.text)
        except json.JSONDecodeError:
            analysis_data = {"raw_response": response.text}

        return {
            "status": "success",
            "file": Path(audio_path).name,
            "analysis": analysis_data
        }

    except Exception as e:
        return {
            "status": "error",
            "message": str(e)
        }

# Example
result = comprehensive_audio_analysis("important_call.mp3")
```

## Frequently Asked Questions

### Q1: What audio formats are supported?

**A:** The API supports common formats including MP3, WAV, M4A, FLAC, OGG, and WebM. Always specify the correct MIME type when sending audio data. If transcription seems poor, try converting to a standard format like MP3 or WAV.

### Q2: Is there a file size limit?

**A:** Yes, individual files are typically limited to 25MB per request. For larger files, either split them into segments or use the Batch API for efficient processing of multiple files.

### Q3: Can the API identify individual speakers in multi-speaker audio?

**A:** The basic transcription doesn't automatically label speakers. However, you can explicitly request speaker identification in your prompt. For production-grade speaker diarization, consider specialized audio processing tools alongside Gemini's content analysis.
