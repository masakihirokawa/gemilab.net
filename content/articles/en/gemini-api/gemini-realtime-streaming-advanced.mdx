# Gemini Advanced Real-Time Streaming — SSE & WebSocket Integration

Gemini's streaming API enables low-latency, real-time applications. This guide covers backend-frontend integration patterns for production-ready implementations.

## Streaming Fundamentals

### Basic Python Streaming

```python
from anthropic import Anthropic

client = Anthropic()

def stream_basic():
    """Basic streaming example"""
    with client.messages.stream(
        model="gemini-2.5-flash",
        max_tokens=1024,
        messages=[{
            "role": "user",
            "content": "Explain asynchronous programming in Python"
        }]
    ) as stream:
        for text in stream.text_stream:
            print(text, end="", flush=True)
    print()

# Execute
stream_basic()
```

<Callout type="info">
Streaming allows tokens to be received in order without waiting for complete response. This dramatically improves perceived response speed.
</Callout>

## Server-Sent Events (SSE) Endpoints

### FastAPI SSE Implementation

```python
from fastapi import FastAPI, Response
from fastapi.responses import StreamingResponse
from anthropic import Anthropic

app = FastAPI()
client = Anthropic()

@app.post("/api/stream")
async def stream_response(request_data: dict):
    """SSE endpoint for streaming responses"""

    async def generate():
        """Stream generator"""
        user_message = request_data.get("message", "")

        try:
            # Start Gemini API streaming
            with client.messages.stream(
                model="gemini-2.5-flash",
                max_tokens=2048,
                messages=[{
                    "role": "user",
                    "content": user_message
                }]
            ) as stream:
                for text in stream.text_stream:
                    # Send in SSE format
                    yield f"data: {text}\n\n"

        except Exception as e:
            yield f"data: ERROR: {str(e)}\n\n"

    return StreamingResponse(
        generate(),
        media_type="text/event-stream",
        headers={
            "Cache-Control": "no-cache",
            "Connection": "keep-alive",
            "X-Accel-Buffering": "no"
        }
    )
```

## WebSocket Integration

### FastAPI WebSocket Implementation

```python
from fastapi import FastAPI, WebSocket
from fastapi.websockets import WebSocketDisconnect
from anthropic import Anthropic
import json

app = FastAPI()
client = Anthropic()

@app.websocket("/ws/chat")
async def websocket_endpoint(websocket: WebSocket):
    """WebSocket chat endpoint"""
    await websocket.accept()

    try:
        while True:
            # Receive message from client
            data = await websocket.receive_text()
            message = json.loads(data)

            # Gemini streaming processing
            response_text = ""
            try:
                with client.messages.stream(
                    model="gemini-2.5-flash",
                    max_tokens=2048,
                    messages=[{
                        "role": "user",
                        "content": message.get("content", "")
                    }]
                ) as stream:
                    for text in stream.text_stream:
                        response_text += text
                        # Send real-time to client
                        await websocket.send_json({
                            "type": "stream",
                            "content": text
                        })

                # Streaming complete
                await websocket.send_json({
                    "type": "complete",
                    "total_tokens": len(response_text.split())
                })

            except Exception as e:
                await websocket.send_json({
                    "type": "error",
                    "message": str(e)
                })

    except WebSocketDisconnect:
        print("Client disconnected")
```

## Chunk Processing

### Processing Chunks into Sentences

```python
import re
from typing import Generator

class StreamChunkProcessor:
    """Process stream chunks into meaningful units"""

    def __init__(self):
        self.buffer = ""
        self.sentence_pattern = re.compile(r'[.!?\n]')

    def process_chunk(self, chunk: str) -> Generator[str, None, None]:
        """Process chunk and yield complete sentences"""
        self.buffer += chunk

        # Split on punctuation
        sentences = self.sentence_pattern.split(self.buffer)

        # Yield complete sentences (exclude last incomplete one)
        for sentence in sentences[:-1]:
            if sentence.strip():
                yield sentence.strip() + "\n"

        # Keep incomplete sentence
        self.buffer = sentences[-1] if sentences[-1] else ""

    def flush(self) -> str:
        """Return remaining buffer content"""
        result = self.buffer
        self.buffer = ""
        return result

# Usage example
processor = StreamChunkProcessor()

with client.messages.stream(
    model="gemini-2.5-flash",
    max_tokens=1024,
    messages=[{"role": "user", "content": "Write a story"}]
) as stream:
    for chunk in stream.text_stream:
        for sentence in processor.process_chunk(chunk):
            print(f"Complete sentence: {sentence}")

    # Process remaining
    final = processor.flush()
    if final:
        print(f"Final part: {final}")
```

## Backpressure Control

### Queue-Based Buffering

```python
import asyncio
from typing import Optional
from datetime import datetime

class StreamBuffer:
    """Stream buffer with backpressure control"""

    def __init__(self, max_size: int = 100):
        self.queue = asyncio.Queue(maxsize=max_size)
        self.done = False

    async def put(self, item: str):
        """Add item to queue with backpressure control"""
        try:
            await asyncio.wait_for(
                self.queue.put(item),
                timeout=5.0
            )
        except asyncio.TimeoutError:
            print("Buffer overflow: production rate too high")

    async def get(self) -> Optional[str]:
        """Get item from queue"""
        if self.done and self.queue.empty():
            return None
        try:
            return await asyncio.wait_for(
                self.queue.get(),
                timeout=5.0
            )
        except asyncio.TimeoutError:
            return None

    def set_done(self):
        """Mark streaming as complete"""
        self.done = True

# Usage example
async def stream_with_backpressure():
    """Streaming with backpressure control"""
    buffer = StreamBuffer(max_size=50)

    async def producer():
        """Producer task"""
        with client.messages.stream(
            model="gemini-2.5-flash",
            max_tokens=2048,
            messages=[{"role": "user", "content": "Generate long text"}]
        ) as stream:
            for chunk in stream.text_stream:
                await buffer.put(chunk)
        buffer.set_done()

    async def consumer():
        """Consumer task"""
        while True:
            item = await buffer.get()
            if item is None:
                break
            print(f"Processing: {item}", end="", flush=True)

    # Concurrent execution
    await asyncio.gather(
        producer(),
        consumer()
    )

# Run
asyncio.run(stream_with_backpressure())
```

## Frontend Integration (TypeScript/JavaScript)

### SSE Consumer

```typescript
// Frontend SSE Client
class GeminiSSEClient {
  private baseUrl: string = "http://localhost:8000";

  async streamMessage(userMessage: string, onChunk: (text: string) => void) {
    try {
      const response = await fetch(`${this.baseUrl}/api/stream`, {
        method: "POST",
        headers: {
          "Content-Type": "application/json",
        },
        body: JSON.stringify({ message: userMessage }),
      });

      if (!response.ok) {
        throw new Error(`HTTP error! status: ${response.status}`);
      }

      const reader = response.body?.getReader();
      if (!reader) throw new Error("No response body");

      const decoder = new TextDecoder();
      let buffer = "";

      while (true) {
        const { done, value } = await reader.read();
        if (done) break;

        buffer += decoder.decode(value, { stream: true });

        // SSE parsing logic
        const lines = buffer.split("\n");
        buffer = lines.pop() || ""; // Keep incomplete line

        for (const line of lines) {
          if (line.startsWith("data: ")) {
            const data = line.slice(6); // Remove "data: "
            if (data !== "[DONE]") {
              onChunk(data);
            }
          }
        }
      }
    } catch (error) {
      console.error("Streaming error:", error);
    }
  }
}

// Usage example
const client = new GeminiSSEClient();
const outputElement = document.getElementById("output");

document.getElementById("sendBtn").addEventListener("click", async () => {
  const input = (document.getElementById("input") as HTMLInputElement).value;
  outputElement!.textContent = ""; // Clear

  await client.streamMessage(input, (chunk) => {
    outputElement!.textContent += chunk;
    outputElement!.scrollTop = outputElement!.scrollHeight;
  });
});
```

### WebSocket Consumer

```typescript
// Frontend WebSocket Client
class GeminiWebSocketClient {
  private ws: WebSocket | null = null;
  private url: string = "ws://localhost:8000/ws/chat";

  connect(onMessage: (data: any) => void) {
    this.ws = new WebSocket(this.url);

    this.ws.onopen = () => {
      console.log("WebSocket connected");
    };

    this.ws.onmessage = (event) => {
      const data = JSON.parse(event.data);
      onMessage(data);
    };

    this.ws.onerror = (error) => {
      console.error("WebSocket error:", error);
    };

    this.ws.onclose = () => {
      console.log("WebSocket disconnected");
    };
  }

  send(message: string) {
    if (this.ws && this.ws.readyState === WebSocket.OPEN) {
      this.ws.send(
        JSON.stringify({
          content: message,
          timestamp: new Date().toISOString(),
        })
      );
    }
  }

  disconnect() {
    if (this.ws) {
      this.ws.close();
    }
  }
}

// Usage example
const wsClient = new GeminiWebSocketClient();
const outputElement = document.getElementById("output");

wsClient.connect((data) => {
  if (data.type === "stream") {
    outputElement!.textContent += data.content;
  } else if (data.type === "complete") {
    outputElement!.textContent += "\n\n[Complete]";
  } else if (data.type === "error") {
    outputElement!.textContent += `\n[Error] ${data.message}`;
  }
});

document.getElementById("sendBtn").addEventListener("click", () => {
  const input = (document.getElementById("input") as HTMLInputElement).value;
  outputElement!.textContent = "";
  wsClient.send(input);
  (document.getElementById("input") as HTMLInputElement).value = "";
});

window.addEventListener("beforeunload", () => {
  wsClient.disconnect();
});
```

## Latency Optimization

### Caching Strategy

```python
from functools import lru_cache
import hashlib
from datetime import datetime

class StreamingCache:
    """Cache streaming responses"""

    def __init__(self):
        self.cache = {}

    def get_cache_key(self, prompt: str) -> str:
        """Use prompt hash as key"""
        return hashlib.md5(prompt.encode()).hexdigest()

    def stream_with_cache(self, model: str, prompt: str, max_age_hours: int = 24):
        """Stream with caching"""
        cache_key = self.get_cache_key(prompt)

        # Check cache
        if cache_key in self.cache:
            cached_response, cached_time = self.cache[cache_key]
            age_hours = (datetime.now() - cached_time).total_seconds() / 3600
            if age_hours < max_age_hours:
                # Return from cache
                for chunk in cached_response.split("\n"):
                    if chunk:
                        yield chunk
                return

        # API request
        response_parts = []
        with client.messages.stream(
            model=model,
            max_tokens=2048,
            messages=[{"role": "user", "content": prompt}]
        ) as stream:
            for text in stream.text_stream:
                response_parts.append(text)
                yield text

        # Cache result
        full_response = "".join(response_parts)
        self.cache[cache_key] = (full_response, datetime.now())

# Usage example
cache = StreamingCache()
for chunk in cache.stream_with_cache("gemini-2.5-flash", "Tell me about Python"):
    print(chunk, end="", flush=True)
```

<Callout type="info">
Caching strategies reduce API calls for repeated requests and dramatically improve response times.
</Callout>

## Error Handling and Recovery

### Retry and Fallback

```python
from tenacity import retry, stop_after_attempt, wait_exponential
import logging

logger = logging.getLogger(__name__)

class RobustStreamingClient:
    """Robust streaming client with error handling"""

    def __init__(self, client, max_retries: int = 3):
        self.client = client
        self.max_retries = max_retries

    @retry(
        stop=stop_after_attempt(3),
        wait=wait_exponential(multiplier=1, min=2, max=10)
    )
    def stream_with_retry(self, model: str, prompt: str):
        """Streaming with retry support"""
        try:
            with self.client.messages.stream(
                model=model,
                max_tokens=2048,
                messages=[{"role": "user", "content": prompt}]
            ) as stream:
                for text in stream.text_stream:
                    yield text
        except Exception as e:
            logger.error(f"Streaming error: {str(e)}")
            raise

    async def stream_with_fallback(self, prompt: str):
        """Streaming with fallback models"""
        models = [
            "gemini-2.5-pro",
            "gemini-2.5-flash",
            "gemini-2.5-flash-lite"
        ]

        for model in models:
            try:
                logger.info(f"Trying model: {model}")
                async for chunk in self.stream_with_retry(model, prompt):
                    yield chunk
                return
            except Exception as e:
                logger.warning(f"Model {model} failed: {str(e)}")
                continue

        raise Exception("All models failed")

# Usage example
robust_client = RobustStreamingClient(client)
for chunk in robust_client.stream_with_retry("gemini-2.5-flash", "Test prompt"):
    print(chunk, end="", flush=True)
```
