## Building Real-time Chat Experiences with Gemini API

Creating engaging user experiences requires two essential capabilities: **streaming responses** that display text progressively, and **multi-turn conversations** that preserve context across exchanges. This guide covers implementing both features with the Gemini API.

## Understanding Streaming Responses

### Standard Non-Streaming Response

First, let's examine the baseline approach without streaming:

```python
from anthropic import Anthropic

client = Anthropic()

# Standard response (wait for complete text)
response = client.messages.create(
    model="gemini-2.5-pro",
    max_tokens=1024,
    messages=[
        {
            "role": "user",
            "content": "Implement quicksort algorithm in Python with detailed comments"
        }
    ]
)

print(response.content[0].text)
```

This works, but users must wait for the entire response before seeing anything.

### Implementing Streaming Responses

Streaming returns text in chunks as it becomes available:

```python
from anthropic import Anthropic

client = Anthropic()

# Enable streaming with context manager
with client.messages.stream(
    model="gemini-2.5-pro",
    max_tokens=1024,
    messages=[
        {
            "role": "user",
            "content": "Implement quicksort algorithm in Python with detailed comments"
        }
    ]
) as stream:
    # Text arrives in real-time
    for text in stream.text_stream:
        print(text, end="", flush=True)
    print()  # Newline after completion

# Access final message and token usage
final_message = stream.get_final_message()
print(f"Total tokens used: {final_message.usage.input_tokens + final_message.usage.output_tokens}")
```

**Key Points**:
- Use `client.messages.stream()` context manager for streaming
- Text arrives in chunks requiring multiple iterations
- `flush=True` ensures immediate display of each chunk
- `get_final_message()` provides complete metrics after streaming completes

## Advanced Chunk Processing

For more sophisticated control over streaming:

```python
from anthropic import Anthropic

client = Anthropic()

def stream_with_tracking(messages, model="gemini-2.5-pro"):
    """Stream responses and track detailed metrics"""
    full_response = ""
    chunk_count = 0
    chunk_sizes = []

    with client.messages.stream(
        model=model,
        max_tokens=2000,
        messages=messages
    ) as stream:
        for text in stream.text_stream:
            full_response += text
            chunk_count += 1
            chunk_sizes.append(len(text))

            # Progress indicator every 100 chunks
            if chunk_count % 100 == 0:
                print(f"[{chunk_count} chunks received...]")

    final_message = stream.get_final_message()
    usage = final_message.usage

    return {
        "response": full_response,
        "chunks": chunk_count,
        "avg_chunk_size": sum(chunk_sizes) / len(chunk_sizes),
        "input_tokens": usage.input_tokens,
        "output_tokens": usage.output_tokens
    }

# Usage example
result = stream_with_tracking([
    {
        "role": "user",
        "content": "Explain the scientific method in comprehensive detail"
    }
])

print(f"Response length: {len(result['response'])} characters")
print(f"Chunks: {result['chunks']}, Avg size: {result['avg_chunk_size']:.1f} chars")
```

## Multi-turn Conversation Implementation

### Building a Chat Session Class

```python
from anthropic import Anthropic

client = Anthropic()

class ChatSession:
    def __init__(self, system_prompt=None):
        self.conversation_history = []
        self.system_prompt = system_prompt or "You are a helpful AI assistant."
        self.model = "gemini-2.5-pro"

    def add_message(self, role, content):
        """Add message to conversation history"""
        self.conversation_history.append({
            "role": role,
            "content": content
        })

    def send_message(self, user_message):
        """Send user message and get response"""
        self.add_message("user", user_message)

        # Send full conversation history to API
        response = client.messages.create(
            model=self.model,
            max_tokens=1024,
            system=self.system_prompt,
            messages=self.conversation_history
        )

        assistant_message = response.content[0].text
        self.add_message("assistant", assistant_message)

        return assistant_message

    def get_history(self):
        """Retrieve full conversation history"""
        return self.conversation_history

    def clear_history(self):
        """Reset conversation"""
        self.conversation_history = []

# Example usage
chat = ChatSession(
    system_prompt="You are an expert Python programmer who explains concepts clearly with code examples."
)

# Turn 1
response1 = chat.send_message("Explain list comprehensions in Python")
print(f"Q: Explain list comprehensions\nA: {response1}\n")

# Turn 2 (maintains context)
response2 = chat.send_message("Show me 5 practical examples")
print(f"Q: Show me 5 practical examples\nA: {response2}\n")

# Turn 3 (still maintains context from previous turns)
response3 = chat.send_message("What about performance benefits?")
print(f"Q: What about performance benefits?\nA: {response3}\n")
```

### Combining Streaming with Multi-turn

```python
class StreamingChatSession(ChatSession):
    """Chat session with streaming support"""

    def send_message_streaming(self, user_message):
        """Send message with streaming response"""
        self.add_message("user", user_message)

        full_response = ""

        with client.messages.stream(
            model=self.model,
            max_tokens=1024,
            system=self.system_prompt,
            messages=self.conversation_history
        ) as stream:
            for text in stream.text_stream:
                full_response += text
                print(text, end="", flush=True)

        print()  # Newline after streaming completes

        self.add_message("assistant", full_response)
        return full_response

# Usage
streaming_chat = StreamingChatSession(
    system_prompt="You are a data science expert. Explain concepts thoroughly with examples."
)

print("Question 1:")
streaming_chat.send_message_streaming("What's the difference between machine learning and deep learning?")

print("\nQuestion 2:")
streaming_chat.send_message_streaming("Tell me more about neural networks")

print("\nQuestion 3:")
streaming_chat.send_message_streaming("How do they work with unstructured data?")
```

## Intelligent Context Management

### Token-Aware History Management

Long conversations accumulate tokens and increase costs:

```python
class ContextAwareChatSession(ChatSession):
    def __init__(self, system_prompt=None, max_context_tokens=4000):
        super().__init__(system_prompt)
        self.max_context_tokens = max_context_tokens

    def estimate_tokens(self, text):
        """Rough token estimation (1 word ≈ 1.3 tokens)"""
        return int(len(text.split()) * 1.3)

    def trim_history(self):
        """Trim old messages if history exceeds token limit"""
        estimated = self.estimate_tokens(str(self.conversation_history))

        while estimated > self.max_context_tokens and len(self.conversation_history) > 1:
            self.conversation_history.pop(0)
            estimated = self.estimate_tokens(str(self.conversation_history))

        return len(self.conversation_history)

    def send_message(self, user_message):
        """Send message with automatic history management"""
        self.add_message("user", user_message)

        messages_kept = self.trim_history()
        print(f"[Context: {messages_kept} messages in history]")

        response = client.messages.create(
            model=self.model,
            max_tokens=1024,
            system=self.system_prompt,
            messages=self.conversation_history
        )

        assistant_message = response.content[0].text
        self.add_message("assistant", assistant_message)

        return assistant_message

# Long conversation with automatic memory management
context_chat = ContextAwareChatSession(
    system_prompt="Helpful assistant",
    max_context_tokens=3000
)

# System automatically trims old messages as needed
for i in range(20):
    response = context_chat.send_message(f"Question {i+1}: Continue our discussion")
```

## Production-Ready Application

### Interactive Coding Assistant

```python
def interactive_coding_assistant():
    """Production-grade coding assistant"""
    chat = StreamingChatSession(
        system_prompt="""
You are a senior software engineer and coding mentor.

Guidelines:
- Provide clean, efficient code examples following best practices
- Explain the reasoning behind your suggestions
- Consider performance, maintainability, and readability
- Help developers understand not just what code does, but why
- Suggest improvements to code posted by users
"""
    )

    print("=== Interactive Coding Assistant ===")
    print("Type 'quit' to exit, 'clear' to reset conversation\n")

    while True:
        user_input = input("You: ").strip()

        if user_input.lower() == 'quit':
            print("Goodbye!")
            break

        if user_input.lower() == 'clear':
            chat.clear_history()
            print("Conversation cleared.\n")
            continue

        if user_input:
            print("Assistant: ", end="")
            chat.send_message_streaming(user_input)
            print()

# Run the assistant
if __name__ == "__main__":
    interactive_coding_assistant()
```

## Best Practices

1. **Strategic Streaming Use**: Enable streaming only when UI needs real-time display
2. **System Prompt Optimization**: Craft system prompts to control behavior reliably
3. **Context Management**: Monitor token usage and prune old messages in long conversations
4. **Error Handling**: Implement retry logic for streaming interruptions
5. **Cost Monitoring**: Track API usage to manage expenses, avoid redundant requests
6. **User Experience**: Display loading indicators while streaming; separate user input from AI responses
