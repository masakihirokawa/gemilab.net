## Gemini API Pricing Structure and Optimization

Gemini API uses Google Cloud's pay-as-you-go model where you pay for actual usage. Managing API costs is the single largest factor determining SaaS profitability.

### Gemini API Pricing (March 2026)

| Model | Input | Output | Best For |
|-------|-------|--------|----------|
| **Gemini 1.5 Flash** | $0.075/1M tokens | $0.30/1M tokens | Fast, lightweight tasks |
| **Gemini 1.5 Pro** | $1.50/1M tokens | $6.00/1M tokens | Complex analysis, long-form generation |
| **Gemini 2.0 Flash** | $0.10/1M tokens | $0.40/1M tokens | Latest model (recommended) |

**Cost Calculation Example**

If a user makes 10 requests daily with average 500 input tokens + 300 output tokens:

```
Daily cost = (10 × 500 × $0.10) / 1,000,000 + (10 × 300 × $0.40) / 1,000,000
           = $0.0005 + $0.0012 = $0.0017/day
           ≈ $0.051/month (per user)

1,000 users: $51/month API cost
```

### API Cost Optimization Strategies

**1. Model Selection Optimization**

Don't use the highest-performing model for every request. Route tasks to appropriate models:

```typescript
// lib/geminiRouter.ts
import { GoogleGenerativeAI } from '@google/generative-ai';

const client = new GoogleGenerativeAI(process.env.GEMINI_API_KEY);

interface TaskRequest {
  prompt: string;
  taskType: 'simple' | 'complex' | 'analysis';
  maxTokens?: number;
}

export async function routeToOptimalModel(
  request: TaskRequest
): Promise<string> {
  let modelName: string;

  switch (request.taskType) {
    case 'simple':
      modelName = 'gemini-2.0-flash'; // Cost: $0.10/$0.40
      break;
    case 'complex':
      modelName = 'gemini-1.5-pro'; // Cost: $1.50/$6.00
      break;
    case 'analysis':
      modelName = 'gemini-1.5-pro';
      break;
  }

  const model = client.getGenerativeModel({ model: modelName });
  const result = await model.generateContent(request.prompt);

  return result.response.text();
}
```

**2. Caching Strategy**

Cache responses to frequently asked questions and templates to reduce API calls:

```typescript
// lib/geminiCache.ts
import Redis from 'redis';
import crypto from 'crypto';

const redis = Redis.createClient();

export async function generateWithCache(
  userId: string,
  prompt: string
): Promise<{ response: string; fromCache: boolean }> {
  const cacheKey = `gemini:${userId}:${hashPrompt(prompt)}`;

  // Check cache (24-hour validity)
  const cached = await redis.get(cacheKey);
  if (cached) {
    return { response: cached, fromCache: true };
  }

  // Call API
  const model = client.getGenerativeModel({ model: 'gemini-2.0-flash' });
  const result = await model.generateContent(prompt);
  const response = result.response.text();

  // Cache result
  await redis.setex(cacheKey, 86400, response);

  return { response, fromCache: false };
}

function hashPrompt(prompt: string): string {
  return crypto.createHash('sha256').update(prompt).digest('hex');
}
```

Process large request volumes as batches during off-peak hours:

```typescript
// lib/batchGemini.ts
export async function processBatch(
  items: Array<{ userId: string; prompt: string }>
) {
  // Prioritize VIP users

      }
    } catch (error) {

    }
  }
}
```

### Stripe Integration Implementation

```typescript
// server/routes/gemini-billing.ts
import Stripe from 'stripe';
import { db } from './db';

const stripe = new Stripe(process.env.STRIPE_SECRET_KEY!);

// Generate monthly invoices based on token usage
export async function generateMonthlyInvoices() {
  const users = await db.getAllActiveUsers();

  for (const user of users) {
    const monthlyUsage = await db.getMonthlyTokenUsage(
      user.id,
      new Date()
    );

    const plan = await db.getUserSubscription(user.id);
    let chargeAmount = 0;

    if (plan.type === 'free') {
      chargeAmount = 0;
    } else if (plan.type === 'standard') {
      chargeAmount = 2900; // $29.00
    } else if (plan.type === 'pro_hybrid') {
      const baseCharge = 4900; // $49.00
      const includedTokens = 5000000;

      if (monthlyUsage.total > includedTokens) {
        const excessTokens = monthlyUsage.total - includedTokens;
        const excessCharge = (excessTokens / 100000) * 50;
        chargeAmount = Math.min(baseCharge + excessCharge, 9900);
      } else {
        chargeAmount = baseCharge;
      }
    }

    if (chargeAmount > 0) {
      const invoice = await stripe.invoices.create({
        customer: user.stripeCustomerId,
        collection_method: 'send_invoice',
        days_until_due: 30,
      });

      await stripe.invoiceItems.create({
        invoice: invoice.id,
        customer: user.stripeCustomerId,
        amount: chargeAmount,
        currency: 'usd',
        description: `Gemini API Usage - ${new Date().toLocaleString('en-US', { month: 'long', year: 'numeric' })}`,
      });

      await stripe.invoices.finalizeInvoice(invoice.id);
    }

    await db.resetMonthlyTokenUsage(user.id);
  }
}

// Track usage on each API request
export async function trackGeminiUsage(
  userId: string,
  inputTokens: number,
  outputTokens: number,
  cost: number
) {
  await db.insertTokenUsage({
    userId,
    inputTokens,
    outputTokens,
    totalTokens: inputTokens + outputTokens,
    cost,
    timestamp: new Date(),
  });

  await db.incrementMonthlyTokenUsage(userId, inputTokens + outputTokens);

  // Alert at 80% of limit
  const plan = await db.getUserSubscription(userId);
  const limit = getTokenLimit(plan.type);
  const current = await db.getMonthlyTokenUsage(userId);

  if (current > limit * 0.8) {
    await notifyUserTokenLimit(userId, current, limit);
  }
}

function getTokenLimit(planType: string): number {
  const limits: { [key: string]: number } = {
    free: 100000,
    standard: 1000000,
    pro_hybrid: 5000000,
  };
  return limits[planType] || 0;
}
```

## Webhook and Subscription Management

Capture Stripe events and sync application state.

### Webhook Endpoint

```typescript
// server/webhooks/stripe-webhook.ts
import Stripe from 'stripe';
import express from 'express';

const stripe = new Stripe(process.env.STRIPE_SECRET_KEY!);
const webhookSecret = process.env.STRIPE_WEBHOOK_SECRET!;

const router = express.Router();

router.post('/webhooks/stripe', (req: any, res: any) => {
  const sig = req.headers['stripe-signature'];
  const buf = req.rawBody;

  let event: Stripe.Event;

  try {
    event = stripe.webhooks.constructEvent(buf, sig, webhookSecret);
  } catch (err) {
    return res.status(400).send(`Webhook Error: ${(err as Error).message}`);
  }

  (async () => {
    try {
      switch (event.type) {
        case 'invoice.payment_succeeded':
          await handleInvoicePaymentSucceeded(
            event.data.object as Stripe.Invoice
          );
          break;

        case 'invoice.payment_failed':
          await handleInvoicePaymentFailed(
            event.data.object as Stripe.Invoice
          );
          break;

        case 'customer.subscription.deleted':
          await handleSubscriptionDeleted(
            event.data.object as Stripe.Subscription
          );
          break;

        case 'customer.subscription.updated':
          await handleSubscriptionUpdated(
            event.data.object as Stripe.Subscription
          );
          break;
      }

      res.json({ received: true });
    } catch (err) {
      console.error('Webhook error:', err);
      res.status(500).json({ error: 'Processing failed' });
    }
  })();
});

async function handleInvoicePaymentSucceeded(invoice: Stripe.Invoice) {
  const userId = invoice.client_reference_id;

  await db.updateUser(userId, {
    lastPaymentAt: new Date(invoice.created * 1000),
    paymentStatus: 'paid',
  });

  await sendEmail(
    await db.getUserEmail(userId),
    'Payment Received',
    `Your payment of $${(invoice.amount_paid / 100).toFixed(2)} has been processed.`
  );
}

async function handleInvoicePaymentFailed(invoice: Stripe.Invoice) {
  const userId = invoice.client_reference_id;
  const retryCount = (await db.getUser(userId)).retryCount + 1;

  await db.updateUser(userId, {
    paymentStatus: 'failed',
    retryCount,
  });

  if (retryCount <= 3) {
    await sendEmail(
      await db.getUserEmail(userId),
      'Payment Failed - Retry Required',
      `Your payment of $${(invoice.amount_due / 100).toFixed(2)} failed. Please update your payment method.`
    );
  } else {
    await suspendUserAccess(userId);
  }
}

async function handleSubscriptionDeleted(subscription: Stripe.Subscription) {
  const userId = subscription.client_reference_id;

  await db.updateUser(userId, {
    plan: 'free',
    subscriptionStatus: 'cancelled',
    cancelledAt: new Date(),
  });

  await revokeAPIAccess(userId);
}

async function handleSubscriptionUpdated(
  subscription: Stripe.Subscription
) {
  const userId = subscription.client_reference_id;
  const newPlan = subscription.items.data[0].price.product;

  await db.updateUser(userId, {
    plan: getPlanNameFromProductId(newPlan),
  });
}

function getPlanNameFromProductId(productId: string): string {
  const mapping: { [key: string]: string } = {
    prod_standard: 'standard',
    prod_pro: 'pro_hybrid',
  };
  return mapping[productId] || 'free';
}
```

## Scaling and Cost Management

Address operational challenges as your SaaS grows.

### Auto-Scaling Implementation

```typescript
// lib/scaling.ts
import * as kubernetes from '@kubernetes/client-node';

const kc = new kubernetes.KubeConfig();
kc.loadFromDefault();

const k8sApi = kc.makeApiClient(kubernetes.AppsV1Api);

export async function autoScaleGeminiWorkers(metrics: {
  queueLength: number;
  avgResponseTime: number;
  errorRate: number;
}) {
  const currentReplicas = await getCurrentDeploymentReplicas(
    'gemini-worker-deployment'
  );

  let targetReplicas = currentReplicas;

  if (metrics.queueLength > 1000) {
    targetReplicas = Math.min(currentReplicas + 5, 50);
  }

  if (metrics.queueLength < 100 && currentReplicas > 5) {
    targetReplicas = Math.max(currentReplicas - 2, 5);
  }

  if (targetReplicas !== currentReplicas) {
    await scaleDeployment(
      'gemini-worker-deployment',
      targetReplicas
    );
  }
}

async function getCurrentDeploymentReplicas(
  deploymentName: string
): Promise<number> {
  const deployment = await k8sApi.readNamespacedDeployment(
    deploymentName,
    'default'
  );
  return deployment.body.spec!.replicas || 1;
}

async function scaleDeployment(
  deploymentName: string,
  replicas: number
): Promise<void> {
  const patch = { spec: { replicas } };
  await k8sApi.patchNamespacedDeployment(
    deploymentName,
    'default',
    patch,
    undefined,
    undefined,
    undefined,
    { headers: { 'Content-Type': 'application/merge-patch+json' } }
  );
}
```
