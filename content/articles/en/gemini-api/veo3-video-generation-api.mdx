## Introduction

**Veo 3** is a state-of-the-art video generation model developed by Google DeepMind. It transforms text prompts and static images into high-quality video content with cinematic precision.

### Veo 3 Core Capabilities

- **Resolution**: Full HD (1080p) video generation
- **Duration**: Up to 8-second video clips
- **Multimodal**: Supports both text prompts and image-to-video animation
- **Audio Integration**: Compatible with embedded audio tracks
- **Gemini Integration**: Accessible via Gemini API for seamless workflows

Within the Google AI ecosystem, Veo 3 enables powerful content creation pipelines. Combined with Gemini 2.0 Flash and higher, you can orchestrate end-to-end workflows: analyze text → generate video → extract metadata.

## Text-to-Video Generation

The most common use case: converting detailed text descriptions into cinematic video. Veo 3 excels at interpreting camera movements, lighting specifications, and visual styles.

### Basic Implementation

```python
import google.generativeai as genai
import os
import time
from pathlib import Path

def generate_video_from_text(prompt: str, output_path: str = "output.mp4"):
    """
    Generate video from a text prompt

    Args:
        prompt: Detailed description of the video to generate
        output_path: Where to save the generated video

    Returns:
        Path to the generated video file
    """

    client = genai.Client(api_key=os.getenv("GOOGLE_API_KEY"))

    print(f"Initiating video generation: {prompt}")

    # Submit async generation request
    response = client.models.generate_video(
        model="models/veo-3",
        prompt=prompt,
        config={
            "duration_seconds": 8,
            "resolution": "1080p",
            "frame_rate": 30
        }
    )

    # Poll until completion (max 15 minutes)
    start_time = time.time()
    max_wait_time = 15 * 60

    while response.state == "PROCESSING":
        elapsed = time.time() - start_time
        if elapsed > max_wait_time:
            raise TimeoutError("Video generation exceeded timeout")

        print(f"Generating... ({elapsed:.0f}s elapsed)")
        time.sleep(10)  # Poll every 10 seconds
        response = client.models.get_generate_video_response(response.id)

    if response.state == "FAILED":
        raise RuntimeError(f"Generation failed: {response.error_message}")

    # Write video to file
    video_data = response.video_bytes

    with open(output_path, "wb") as f:
        f.write(video_data)

    print(f"Video saved to: {output_path}")
    return output_path


# Example usage
if __name__ == "__main__":
    prompt = """
    Cinematic wide shot of a serene Japanese garden at sunrise.
    A stone lantern sits peacefully near a still pond, reflecting
    the golden morning light. Cherry blossoms gently fall from above.
    Slow dolly move forward, gentle 24fps cinematic motion.
    Color grade: warm, soft focus edges, cinematic depth of field.
    """

    generate_video_from_text(prompt, "japan_garden.mp4")
```

## Image-to-Video Animation

Transform static images into animated video by specifying desired motion. Perfect for product photography, illustrations, and architectural renders.

### Implementation Example

```python
import google.generativeai as genai
from pathlib import Path
import base64
import time
import os

def animate_image_to_video(
    image_path: str,
    motion_prompt: str,
    output_path: str = "animated.mp4"
):
    """
    Animate a static image into video

    Args:
        image_path: Path to input image file
        motion_prompt: Desired motion and animation description
        output_path: Output video file path

    Returns:
        Path to generated animation
    """

    client = genai.Client(api_key=os.getenv("GOOGLE_API_KEY"))

    # Encode image to base64
    image_data = Path(image_path).read_bytes()
    image_b64 = base64.b64encode(image_data).decode('utf-8')

    # Infer MIME type from extension
    extension = Path(image_path).suffix.lower()
    mime_types = {
        ".jpg": "image/jpeg",
        ".jpeg": "image/jpeg",
        ".png": "image/png",
        ".gif": "image/gif",
        ".webp": "image/webp"
    }
    mime_type = mime_types.get(extension, "image/jpeg")

    print(f"Uploading image: {image_path}")

    # Submit image animation request
    response = client.models.generate_video(
        model="models/veo-3",
        image_source=image_b64,
        mime_type=mime_type,
        prompt=motion_prompt,
        config={
            "duration_seconds": 4,
            "resolution": "1080p",
            "animation_style": "smooth",
            "frame_rate": 24
        }
    )

    # Poll for completion
    start_time = time.time()
    max_wait = 20 * 60

    while response.state == "PROCESSING":
        elapsed = time.time() - start_time
        if elapsed > max_wait:
            raise TimeoutError("Image animation generation timed out")

        percentage = response.progress_percent
        print(f"Progress: {percentage}% ({elapsed:.0f}s elapsed)")
        time.sleep(5)
        response = client.models.get_generate_video_response(response.id)

    if response.state == "FAILED":
        raise RuntimeError(f"Generation failed: {response.error_message}")

    # Save animation to file
    with open(output_path, "wb") as f:
        f.write(response.video_bytes)

    print(f"Animation saved: {output_path}")
    return output_path


# Example usage
if __name__ == "__main__":
    motion_prompt = """
    The landscape comes alive with subtle motion.
    Wind gently moves grass and tree foliage.
    Clouds slowly drift across the sky at different speeds.
    Camera perspective remains static and centered.
    Preserve original composition and color palette.
    Duration: 4 seconds of smooth, natural motion.
    """

    animate_image_to_video(
        "landscape.png",
        motion_prompt,
        "animated_landscape.mp4"
    )
```

<Callout type="info">
Image-to-video preserves composition and color, making it ideal for maintaining visual identity while adding motion to existing assets.

```python
class VideoPromptBuilder:
    """Production-grade prompt composition"""

    def __init__(self):
        self.sections = {}

    def set_scene(self, scene_description: str):
        """Main scene setting"""
        self.sections['scene'] = scene_description
        return self

    def set_camera_motion(self, camera_work: str):
        """Camera movement specification"""
        self.sections['camera'] = camera_work
        return self

    def set_lighting(self, lighting_style: str):
        """Lighting and color grading"""
        self.sections['lighting'] = lighting_style
        return self

    def set_style(self, visual_style: str):
        """Overall visual aesthetic"""
        self.sections['style'] = visual_style
        return self

    def set_duration(self, seconds: int):
        """Video duration in seconds"""
        self.sections['duration'] = f"{seconds} seconds"
        return self

    def add_negative_prompt(self, negative: str):
        """Elements to avoid"""
        self.sections['negative'] = negative
        return self

    def build(self) -> str:
        """Compile final prompt"""
        parts = []

        if 'scene' in self.sections:
            parts.append(f"Scene: {self.sections['scene']}")
        if 'camera' in self.sections:
            parts.append(f"Camera: {self.sections['camera']}")
        if 'lighting' in self.sections:
            parts.append(f"Lighting: {self.sections['lighting']}")
        if 'style' in self.sections:
            parts.append(f"Style: {self.sections['style']}")
        if 'duration' in self.sections:
            parts.append(f"Duration: {self.sections['duration']}")
        if 'negative' in self.sections:
            parts.append(f"Avoid: {self.sections['negative']}")

        return "\n".join(parts)
```

## Async Processing and Batch Generation

Production-grade video generation requires efficient handling of concurrent requests and long-running operations.

### Async Polling Pattern

```python
import asyncio
import google.generativeai as genai
from typing import List
import os
from dataclasses import dataclass
from datetime import datetime

@dataclass
class GenerationJob:
    """Tracks video generation job lifecycle"""
    job_id: str
    prompt: str
    output_path: str
    created_at: datetime
    status: str = "PENDING"
    progress: int = 0
    error: str = None

class VideoGenerationQueue:
    """Async video generation queue manager"""

    def __init__(self, api_key: str, max_concurrent: int = 5):
        self.client = genai.Client(api_key=api_key)
        self.max_concurrent = max_concurrent
        self.jobs: dict[str, GenerationJob] = {}
        self.semaphore = asyncio.Semaphore(max_concurrent)

    async def submit_job(
        self,
        prompt: str,
        output_path: str
    ) -> str:
        """Submit a video generation job"""
        response = self.client.models.generate_video(
            model="models/veo-3",
            prompt=prompt,
            config={
                "duration_seconds": 8,
                "resolution": "1080p"
            }
        )

        job = GenerationJob(
            job_id=response.id,
            prompt=prompt,
            output_path=output_path,
            created_at=datetime.now(),
            status="PROCESSING"
        )

        self.jobs[response.id] = job

        asyncio.create_task(
            self._poll_job(response.id)
        )

        return response.id

    async def _poll_job(self, job_id: str):
        """Poll job status until completion"""
        async with self.semaphore:
            job = self.jobs[job_id]
            max_wait = 20 * 60  # 20 minutes
            elapsed = 0
            poll_interval = 10  # 10 seconds

            while elapsed < max_wait:
                try:
                    response = self.client.models.get_generate_video_response(
                        job_id
                    )

                    job.progress = response.progress_percent

                    if response.state == "COMPLETED":
                        with open(job.output_path, "wb") as f:
                            f.write(response.video_bytes)
                        job.status = "COMPLETED"
                        print(f"✓ Completed: {job_id}")
                        return

                    elif response.state == "FAILED":
                        job.status = "FAILED"
                        job.error = response.error_message
                        print(f"✗ Failed: {job_id} - {response.error_message}")
                        return

                    await asyncio.sleep(poll_interval)
                    elapsed += poll_interval

                except Exception as e:
                    job.status = "ERROR"
                    job.error = str(e)
                    print(f"Error: {job_id} - {e}")
                    return

            job.status = "TIMEOUT"
            job.error = "Generation exceeded timeout"

    async def wait_for_completion(
        self,
        job_ids: List[str],
        timeout_seconds: int = 1200
    ) -> dict[str, GenerationJob]:
        """Wait for multiple jobs to complete"""
        start = datetime.now()

        while True:
            elapsed = (datetime.now() - start).total_seconds()
            if elapsed > timeout_seconds:
                break

            all_done = all(
                self.jobs[jid].status in ["COMPLETED", "FAILED", "TIMEOUT"]
                for jid in job_ids
            )

            if all_done:
                break

            await asyncio.sleep(5)

        return {
            jid: self.jobs[jid]
            for jid in job_ids
        }


# Usage example
async def batch_generate_videos():
    """Generate multiple videos concurrently"""
    queue = VideoGenerationQueue(
        api_key=os.getenv("GOOGLE_API_KEY"),
        max_concurrent=3
    )

    prompts = [
        ("Sunset over ocean waves, warm golden light", "sunset.mp4"),
        ("Futuristic city skyline with flying vehicles", "city.mp4"),
        ("Forest canopy with dappled sunlight", "forest.mp4"),
    ]

    job_ids = []
    for prompt, output in prompts:
        job_id = await queue.submit_job(prompt, output)
        job_ids.append(job_id)

    results = await queue.wait_for_completion(job_ids, timeout_seconds=1800)

    for job_id, job in results.items():
        print(f"{job_id}: {job.status} - {job.output_path}")


if __name__ == "__main__":
    asyncio.run(batch_generate_videos())
```

### API Route Handler

```typescript
import { GoogleGenerativeAI } from "@google/generative-ai";
import { NextRequest, NextResponse } from "next/server";
import fs from "fs/promises";
import path from "path";

const client = new GoogleGenerativeAI(
  process.env.GOOGLE_API_KEY || ""
);

export async function POST(request: NextRequest) {
  try {
    const body = await request.json();
    const { prompt, outputFilename } = body;

    if (!prompt) {
      return NextResponse.json(
        { error: "prompt is required" },
        { status: 400 }
      );
    }

    console.log("Starting video generation:", prompt);

    // Submit generation request
    const response = await client.models.generateVideo({
      model: "models/veo-3",
      prompt: prompt,
      config: {
        duration_seconds: 8,
        resolution: "1080p",
        frame_rate: 30,
      },
    });

    let currentResponse = response;
    const maxWaitTime = 15 * 60 * 1000; // 15 minutes
    const startTime = Date.now();

    // Poll until completion
    while (currentResponse.state === "PROCESSING") {
      const elapsed = Date.now() - startTime;

      if (elapsed > maxWaitTime) {
        return NextResponse.json(
          { error: "Video generation timeout" },
          { status: 408 }
        );
      }

      console.log(
        `Progress: ${currentResponse.progress_percent}% (${elapsed / 1000}s)`
      );

      await new Promise((resolve) => setTimeout(resolve, 10000));

      currentResponse = await client.models.getGenerateVideoResponse(
        currentResponse.id
      );
    }

    if (currentResponse.state === "FAILED") {
      return NextResponse.json(
        { error: `Generation failed: ${currentResponse.error_message}` },
        { status: 500 }
      );
    }

    // Save video to file
    const videosDir = path.join(process.cwd(), "public", "videos");
    await fs.mkdir(videosDir, { recursive: true });

    const filename =
      outputFilename ||
      `video_${Date.now()}.mp4`;
    const filepath = path.join(videosDir, filename);

    if (currentResponse.video_bytes) {
      const buffer = Buffer.from(currentResponse.video_bytes);
      await fs.writeFile(filepath, buffer);
    }

    return NextResponse.json(
      {
        success: true,
        videoPath: `/videos/${filename}`,
        message: "Video generated successfully",
      },
      { status: 200 }
    );
  } catch (error) {
    console.error("Error:", error);
    return NextResponse.json(
      {
        error: error instanceof Error ? error.message : "Unknown error",
      },
      { status: 500 }
    );
  }
}
```

#### 2. Caching Generated Videos

```python
import hashlib
import json
import os

class VideoCache:
    """Cache generated videos by prompt hash"""

    def __init__(self, cache_dir: str = ".cache/videos"):
        self.cache_dir = cache_dir
        os.makedirs(cache_dir, exist_ok=True)

    def get_cache_key(self, prompt: str, config: dict) -> str:
        """Generate cache key from prompt and config"""
        cache_data = json.dumps({
            "prompt": prompt,
            "config": config
        }, sort_keys=True)

        return hashlib.sha256(cache_data.encode()).hexdigest()

    def get_cached_video(self, prompt: str, config: dict) -> str | None:
        """Retrieve cached video if available"""
        key = self.get_cache_key(prompt, config)
        cache_file = f"{self.cache_dir}/{key}.mp4"

        if os.path.exists(cache_file):
            print(f"Cache hit: {key}")
            return cache_file

        return None

    def save_to_cache(self, prompt: str, config: dict, video_path: str):
        """Store generated video in cache"""
        key = self.get_cache_key(prompt, config)
        cache_file = f"{self.cache_dir}/{key}.mp4"

        import shutil
        shutil.copy(video_path, cache_file)
        print(f"Cached: {cache_file}")
```

```python
def batch_generate_with_budget(
    prompts: List[tuple[str, dict]],
    budget_usd: float = 50.0
) -> List[str]:
    """Generate videos within budget constraints"""

    cost_map = {
        "480p": 0.10,
        "720p": 0.25,
        "1080p": 0.60
    }

    remaining_budget = budget_usd
    results = []

    for prompt, config in prompts:

        output = generate_video_from_text(prompt, config)
        results.append(output)
        remaining_budget -= cost

        print(f"Remaining budget: ${remaining_budget:.2f}")

    return results
```

## Error Handling and Debugging

### Error Type Classification

```python
from enum import Enum
from typing import Optional

class GenerationError(Enum):
    """Video generation error taxonomy"""
    INVALID_API_KEY = "Authentication failed"
    QUOTA_EXCEEDED = "Monthly quota exceeded"
    INVALID_PROMPT = "Prompt contains prohibited content"
    TIMEOUT = "Generation exceeded max wait time"
    NETWORK_ERROR = "Network or connectivity issue"
    UNSUPPORTED_FORMAT = "Image format not supported"

def diagnose_error(error: Exception) -> Optional[GenerationError]:
    """Map exceptions to error types"""
    error_str = str(error).lower()

    if "authentication" in error_str or "unauthorized" in error_str:
        return GenerationError.INVALID_API_KEY
    elif "quota" in error_str or "limit" in error_str:
        return GenerationError.QUOTA_EXCEEDED
    elif "invalid prompt" in error_str:
        return GenerationError.INVALID_PROMPT
    elif "timeout" in error_str:
        return GenerationError.TIMEOUT
    elif "connection" in error_str:
        return GenerationError.NETWORK_ERROR
    elif "format" in error_str:
        return GenerationError.UNSUPPORTED_FORMAT

    return None
```
