# recipe_04_image_caption.py — 画像キャプション＋Altテキスト生成
import pathlib
from common_setup import client, MODEL

def generate_image_caption(image_path: str) -> str:
    """画像からキャプションとAltテキストを生成する"""
    image_data = pathlib.Path(image_path).read_bytes()
    ext = pathlib.Path(image_path).suffix.lower()
    mime_map = {".png": "image/png", ".jpg": "image/jpeg", ".jpeg": "image/jpeg", ".webp": "image/webp"}
    mime_type = mime_map.get(ext, "image/jpeg")

    response = client.models.generate_content(
        model=MODEL,
        contents=[
            {
                "role": "user",
                "parts": [
                    {"inline_data": {"mime_type": mime_type, "data": image_data}},
                    {"text": """この画像について以下を生成してください:
1. SNS投稿用キャプション（日本語、50文字以内）
2. ウェブサイト用Altテキスト（日本語、100文字以内）
3. SEO用画像説明（日本語、150文字以内）
4. 推奨ハッシュタグ（5個）"""}
                ]
            }
        ]
    )
    return response.text

## レシピ10: バッチ処理 — 複数ファイルの一括変換

フォルダ内の複数ファイルを一括で処理するバッチスクリプトです。

```python
# recipe_10_batch_processing.py — 複数ファイル一括処理
import pathlib
import json
from common_setup import client, MODEL

def batch_process_files(
    input_dir: str,
    output_dir: str,
    task: str = "summarize"
) -> list[dict]:
    """フォルダ内のテキストファイルを一括処理する"""
    input_path = pathlib.Path(input_dir)
    output_path = pathlib.Path(output_dir)
    output_path.mkdir(parents=True, exist_ok=True)

    task_prompts = {
        "summarize": "以下のテキストを200文字以内で要約してください。",
        "translate": "以下のテキストを英語に翻訳してください。",
        "keywords": "以下のテキストから重要なキーワードを10個抽出してください。",
    }
    prompt_template = task_prompts.get(task, task)
    results = []

    for file in sorted(input_path.glob("*.txt")):
        content = file.read_text(encoding="utf-8")
        prompt = f"{prompt_template}\n\n---\n{content}"

        response = client.models.generate_content(
            model=MODEL,
            contents=prompt
        )

        output_file = output_path / f"{file.stem}_{task}.txt"
        output_file.write_text(response.text, encoding="utf-8")
        results.append({"file": file.name, "status": "success"})
        print(f"✅ {file.name} → {output_file.name}")

    # 処理結果のサマリーを保存
    summary_file = output_path / "batch_summary.json"
    summary_file.write_text(
        json.dumps(results, ensure_ascii=False, indent=2),
        encoding="utf-8"
    )
    return results

# 使用例
# batch_process_files("./input_docs", "./output_docs", task="summarize")
# 期待する出力例:
# ✅ report_01.txt → report_01_summarize.txt
# ✅ report_02.txt → report_02_summarize.txt
# ✅ report_03.txt → report_03_summarize.txt
```

## よくある質問（FAQ）

### Q. Gemini APIの無料枠で、これらのレシピはすべて使えますか？

はい、すべてのレシピは無料枠の範囲内で利用できます。Gemini 2.5 Flashは無料枠でも十分なリクエスト数が確保されており、個人利用や学習目的であれば問題ありません。大量のバッチ処理を行う場合は、[APIのレート制限とクォータ管理ガイド](/articles/gemini-api/gemini-api-rate-limiting-quota-management)をご確認ください。

### Q. レシピのコードを本番環境で使う際の注意点は？

本番環境で使用する際は、エラーハンドリングの追加、APIキーの安全な管理（環境変数やシークレットマネージャーの使用）、そしてレート制限への対応が必要です。また、APIレスポンスのバリデーションも実装することをお勧めします。本番運用の詳細は[Gemini APIコスト最適化ガイド](/articles/gemini-api/gemini-api-cost-optimization)を参照してください。

### Q. 他のプログラミング言語でも使えますか？

Gemini APIはREST APIとして提供されているため、どの言語からでも利用可能です。公式SDKはPython、JavaScript/TypeScript、Go、Dartに対応しています。この記事ではPythonを使用していますが、ロジックは他の言語にも応用できます。

### Q. 生成結果の品質を向上させるコツはありますか？

プロンプトの具体性を高めることが最も効果的です。出力形式を明確に指定する、具体例を含める、制約条件を記述するなどの工夫で品質が大幅に向上します。詳しくは[Gemini プロンプトエンジニアリングガイド](/articles/gemini-api/gemini-prompt-engineering-guide)をお読みください。
