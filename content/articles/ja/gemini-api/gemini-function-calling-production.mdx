## はじめに：Function Calling + 構造化出力の組み合わせ

Gemini の **Function Calling** と **responseSchema（構造化出力）** は別々の機能ですが、組み合わせることで強力な型安全なAIパイプラインを構築できます。

```
Function Calling: LLM が外部ツール/APIを呼び出す機能
      ↕
responseSchema: LLM の出力を JSON Schema に強制する機能
```

本記事では実際の本番システムで使えるパターンを Python + TypeScript の両方で実装します。

<Callout type="info">
Google AI SDK (`google-generativeai` / `@google/generative-ai`) が必要です。`GOOGLE_API_KEY` を環境変数に設定してください。
</Callout>

## 1. 基本的な Function Calling

### Python 実装

```python
# function_calling_basic.py
import google.generativeai as genai
import json
from typing import Any

genai.configure(api_key="YOUR_API_KEY")

# ツール定義
WEATHER_TOOL = genai.protos.Tool(
    function_declarations=[
        genai.protos.FunctionDeclaration(
            name="get_weather",
            description="指定した都市の現在の天気を取得する",
            parameters=genai.protos.Schema(
                type=genai.protos.Type.OBJECT,
                properties={
                    "city": genai.protos.Schema(
                        type=genai.protos.Type.STRING,
                        description="都市名（例: Tokyo, Osaka）",
                    ),
                    "unit": genai.protos.Schema(
                        type=genai.protos.Type.STRING,
                        enum=["celsius", "fahrenheit"],
                        description="温度の単位",
                    ),
                },
                required=["city"],
            ),
        )
    ]
)

# ツール実装（実際はAPIを呼ぶ）
def get_weather(city: str, unit: str = "celsius") -> dict:
    # 本番では Open-Meteo や OpenWeatherMap API を呼ぶ
    mock_data = {
        "Tokyo": {"temp": 22, "condition": "晴れ", "humidity": 60},
        "Osaka": {"temp": 24, "condition": "曇り", "humidity": 65},
    }
    data = mock_data.get(city, {"temp": 20, "condition": "不明", "humidity": 50})
    temp = data["temp"]
    if unit == "fahrenheit":
        temp = temp * 9 / 5 + 32
    return {**data, "temp": temp, "unit": unit, "city": city}

# ツールディスパッチャー
TOOL_REGISTRY = {
    "get_weather": get_weather,
}

def dispatch_tool_call(function_call) -> str:
    func_name = function_call.name
    args = dict(function_call.args)

    if func_name not in TOOL_REGISTRY:
        return json.dumps({"error": f"未知のツール: {func_name}"})

    try:
        result = TOOL_REGISTRY[func_name](**args)
        return json.dumps(result, ensure_ascii=False)
    except Exception as e:
        return json.dumps({"error": str(e)})

# エージェントループ
def run_with_tools(user_message: str) -> str:
    model = genai.GenerativeModel(
        model_name="gemini-2.0-flash",
        tools=[WEATHER_TOOL],
    )

    chat = model.start_chat()
    response = chat.send_message(user_message)

    # ツール呼び出しループ
    while response.candidates[0].content.parts:
        parts = response.candidates[0].content.parts
        function_calls = [p for p in parts if hasattr(p, "function_call")]

        if not function_calls:
            break

        # ツール結果を返す
        tool_responses = []
        for part in function_calls:
            result = dispatch_tool_call(part.function_call)
            tool_responses.append(
                genai.protos.Part(
                    function_response=genai.protos.FunctionResponse(
                        name=part.function_call.name,
                        response={"result": json.loads(result)},
                    )
                )
            )

        response = chat.send_message(tool_responses)

    return response.text

# 使用例
if __name__ == "__main__":
    result = run_with_tools("東京と大阪の今日の天気を比較してください")
    print(result)
```

## 2. 構造化出力（responseSchema）

```python
# structured_output.py
import google.generativeai as genai
import json
from typing import TypedDict

genai.configure(api_key="YOUR_API_KEY")

# 出力スキーマの定義
ARTICLE_SCHEMA = {
    "type": "object",
    "properties": {
        "title": {
            "type": "string",
            "description": "記事タイトル（30文字以内）"
        },
        "summary": {
            "type": "string",
            "description": "記事の要約（100文字以内）"
        },
        "sentiment": {
            "type": "string",
            "enum": ["positive", "negative", "neutral"],
            "description": "記事の感情トーン"
        },
        "topics": {
            "type": "array",
            "items": {"type": "string"},
            "description": "記事のトピックタグ（最大5つ）"
        },
        "difficulty": {
            "type": "integer",
            "minimum": 1,
            "maximum": 5,
            "description": "技術的難易度（1=初心者 〜 5=エキスパート）"
        }
    },
    "required": ["title", "summary", "sentiment", "topics", "difficulty"]
}

def analyze_article(content: str) -> dict:
    model = genai.GenerativeModel(
        model_name="gemini-2.0-flash",
        generation_config=genai.GenerationConfig(
            response_mime_type="application/json",
            response_schema=ARTICLE_SCHEMA,
        ),
    )

    response = model.generate_content(
        f"以下の記事を分析してください:\n\n{content}"
    )

    # レスポンスは保証された JSON
    return json.loads(response.text)

# Python TypedDict との連携
class ArticleAnalysis(TypedDict):
    title: str
    summary: str
    sentiment: str
    topics: list[str]
    difficulty: int

def analyze_article_typed(content: str) -> ArticleAnalysis:
    result = analyze_article(content)
    return ArticleAnalysis(**result)
```

## 3. 並列 Function Calling（高度な実装）

```python
# parallel_function_calling.py
import asyncio
import json
import google.generativeai as genai
from google.generativeai.types import FunctionLibrary

# 複数ツールの並列実行をサポートするエージェント
class ParallelToolAgent:
    def __init__(self, tools: list, model_name: str = "gemini-2.0-flash"):
        self.model = genai.GenerativeModel(
            model_name=model_name,
            tools=tools,
            tool_config={"function_calling_config": {"mode": "AUTO"}},
        )
        self.tool_registry: dict = {}

    def register_tool(self, name: str, func):
        self.tool_registry[name] = func

    async def _call_tool_async(self, function_call) -> genai.protos.Part:
        """非同期でツールを実行"""
        func_name = function_call.name
        args = dict(function_call.args)

        if func_name not in self.tool_registry:
            result = {"error": f"未知のツール: {func_name}"}
        else:
            try:
                func = self.tool_registry[func_name]
                # async 関数と同期関数の両方をサポート
                if asyncio.iscoroutinefunction(func):
                    result = await func(**args)
                else:
                    result = await asyncio.get_event_loop().run_in_executor(
                        None, lambda: func(**args)
                    )
            except Exception as e:
                result = {"error": str(e)}

        return genai.protos.Part(
            function_response=genai.protos.FunctionResponse(
                name=func_name,
                response={"result": result},
            )
        )

    async def run(self, user_message: str, max_iterations: int = 10) -> str:
        chat = self.model.start_chat()
        response = await asyncio.get_event_loop().run_in_executor(
            None, lambda: chat.send_message(user_message)
        )

        for _ in range(max_iterations):
            parts = response.candidates[0].content.parts
            function_calls = [p for p in parts if hasattr(p, "function_call") and p.function_call.name]

            if not function_calls:
                break

            # 全ツール呼び出しを並列実行 ← ここがポイント
            tool_results = await asyncio.gather(
                *[self._call_tool_async(p.function_call) for p in function_calls]
            )

            response = await asyncio.get_event_loop().run_in_executor(
                None, lambda: chat.send_message(list(tool_results))
            )

        return response.text


# 実用例：株価 + ニュース + 為替を並列取得
import aiohttp

async def get_stock_price(symbol: str) -> dict:
    # 実際は Yahoo Finance API など
    await asyncio.sleep(0.5)  # API レイテンシシミュレーション
    return {"symbol": symbol, "price": 150.25, "change": "+2.3%"}

async def get_latest_news(topic: str) -> dict:
    await asyncio.sleep(0.3)
    return {"topic": topic, "headlines": [f"{topic} の最新ニュース1", f"{topic} の最新ニュース2"]}

async def get_exchange_rate(from_currency: str, to_currency: str) -> dict:
    await asyncio.sleep(0.2)
    return {"from": from_currency, "to": to_currency, "rate": 149.5}


async def main():
    TOOLS = [
        genai.protos.Tool(function_declarations=[
            genai.protos.FunctionDeclaration(
                name="get_stock_price",
                description="株価を取得",
                parameters=genai.protos.Schema(
                    type=genai.protos.Type.OBJECT,
                    properties={"symbol": genai.protos.Schema(type=genai.protos.Type.STRING)},
                    required=["symbol"],
                ),
            ),
            genai.protos.FunctionDeclaration(
                name="get_latest_news",
                description="最新ニュースを取得",
                parameters=genai.protos.Schema(
                    type=genai.protos.Type.OBJECT,
                    properties={"topic": genai.protos.Schema(type=genai.protos.Type.STRING)},
                    required=["topic"],
                ),
            ),
            genai.protos.FunctionDeclaration(
                name="get_exchange_rate",
                description="為替レートを取得",
                parameters=genai.protos.Schema(
                    type=genai.protos.Type.OBJECT,
                    properties={
                        "from_currency": genai.protos.Schema(type=genai.protos.Type.STRING),
                        "to_currency": genai.protos.Schema(type=genai.protos.Type.STRING),
                    },
                    required=["from_currency", "to_currency"],
                ),
            ),
        ])
    ]

    agent = ParallelToolAgent(tools=TOOLS)
    agent.register_tool("get_stock_price", get_stock_price)
    agent.register_tool("get_latest_news", get_latest_news)
    agent.register_tool("get_exchange_rate", get_exchange_rate)

    result = await agent.run(
        "NVDA の株価、AI業界の最新ニュース、USD/JPY の為替レートを一度に調べて日本語でまとめてください"
    )
    print(result)

asyncio.run(main())
```

## 4. TypeScript 実装（Next.js API Route 版）

```typescript
// app/api/gemini-agent/route.ts
import { GoogleGenerativeAI, FunctionCallingMode } from '@google/generative-ai';
import { NextRequest, NextResponse } from 'next/server';

const genai = new GoogleGenerativeAI(process.env.GOOGLE_API_KEY!);

// ツール定義
const TOOLS = [
  {
    functionDeclarations: [
      {
        name: 'search_products',
        description: '商品を検索する',
        parameters: {
          type: 'object' as const,
          properties: {
            query: { type: 'string', description: '検索クエリ' },
            maxResults: { type: 'number', description: '最大件数（デフォルト5）' },
          },
          required: ['query'],
        },
      },
      {
        name: 'get_product_details',
        description: '商品の詳細情報を取得する',
        parameters: {
          type: 'object' as const,
          properties: {
            productId: { type: 'string', description: '商品ID' },
          },
          required: ['productId'],
        },
      },
    ],
  },
];

// ツール実装
async function searchProducts(query: string, maxResults = 5): Promise<object> {
  // 実際はデータベースを検索
  return {
    results: Array.from({ length: maxResults }, (_, i) => ({
      id: `prod_${i + 1}`,
      name: `${query} 商品 ${i + 1}`,
      price: (1000 + i * 500),
    })),
  };
}

async function getProductDetails(productId: string): Promise<object> {
  return {
    id: productId,
    name: `商品 ${productId}`,
    price: 1500,
    description: '詳細な商品説明...',
    stock: 42,
  };
}

const TOOL_DISPATCH: Record<string, (...args: any[]) => Promise<object>> = {
  search_products: ({ query, maxResults }: { query: string; maxResults?: number }) =>
    searchProducts(query, maxResults),
  get_product_details: ({ productId }: { productId: string }) =>
    getProductDetails(productId),
};

export async function POST(req: NextRequest) {
  const { message } = await req.json();

  const model = genai.getGenerativeModel({
    model: 'gemini-2.0-flash',
    tools: TOOLS,
    toolConfig: { functionCallingConfig: { mode: FunctionCallingMode.AUTO } },
  });

  const chat = model.startChat();
  let response = await chat.sendMessage(message);

  // ツールループ
  for (let i = 0; i < 10; i++) {
    const functionCalls = response.response.functionCalls();
    if (!functionCalls || functionCalls.length === 0) break;

    // 並列実行
    const results = await Promise.all(
      functionCalls.map(async (fc) => {
        const func = TOOL_DISPATCH[fc.name];
        const result = func ? await func(fc.args as any) : { error: `未知のツール: ${fc.name}` };
        return { functionResponse: { name: fc.name, response: result } };
      }),
    );

    response = await chat.sendMessage(results);
  }

  return NextResponse.json({ result: response.response.text() });
}
```

## まとめ

Gemini の Function Calling + 構造化出力を組み合わせることで、**型安全で本番グレードのAI APIパイプライン** が構築できます。

重要なポイントをまとめます：

- **並列 Function Calling** — 複数ツールを同時実行してレイテンシを削減
- **responseSchema** — 出力を JSON Schema で強制し、パースエラーを排除
- **asyncio** — Python での非同期ツール実行で I/O ボトルネックを解消
- **モデル選択** — タスク複雑度に応じて Flash/Pro を使い分けてコスト最適化
