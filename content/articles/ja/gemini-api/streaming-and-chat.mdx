## Gemini APIでリアルタイムなチャット体験を実装する

ユーザーエクスペリエンスの向上には、テキストが段階的に表示される**ストリーミング応答**と、会話の前後関係を保持する**マルチターン会話**の実装が重要です。このガイドでは、Gemini APIを使ってこれらの機能を実装する方法を詳しく解説します。

## ストリーミング応答の基礎

### 標準的な非ストリーミング応答

まず、ストリーミング以前の標準的な実装を確認しましょう：

```python
from anthropic import Anthropic

client = Anthropic()

# 標準的な応答（全文が返されるまで待機）
response = client.messages.create(
    model="gemini-2.5-pro",
    max_tokens=1024,
    messages=[
        {
            "role": "user",
            "content": "Pythonでクイックソートを実装してください"
        }
    ]
)

print(response.content[0].text)
```

このアプローチは動作しますが、長い応答の場合ユーザーは完了まで待つ必要があります。

### ストリーミング応答の実装

ストリーミングを有効にすると、テキストがチャンク単位でリアルタイムに返されます：

```python
from anthropic import Anthropic

client = Anthropic()

# ストリーミングを有効にした実装
with client.messages.stream(
    model="gemini-2.5-pro",
    max_tokens=1024,
    messages=[
        {
            "role": "user",
            "content": "Pythonでクイックソートを実装してください"
        }
    ]
) as stream:
    # テキストがリアルタイムで返される
    for text in stream.text_stream:
        print(text, end="", flush=True)
    print()  # 改行

print(f"\nUsage: {stream.get_final_message().usage}")
```

**重要なポイント**:
- `stream=True` または `client.messages.stream()` でストリーミングを有効化
- テキストはチャンク単位で返されるため、複数回のイテレーションが発生
- `flush=True`により、各チャンク出力後すぐに表示を更新
- `get_final_message()` で最終メッセージとトークン使用量を取得可能

## チャンク処理とトークン数の追跡

ストリーミング中により詳細な制御が必要な場合：

```python
from anthropic import Anthropic

client = Anthropic()

def stream_and_collect(messages, model="gemini-2.5-pro"):
    """ストリーミング応答を収集し、トークン数を追跡する"""
    full_response = ""
    chunk_count = 0

    with client.messages.stream(
        model=model,
        max_tokens=2000,
        messages=messages
    ) as stream:
        for text in stream.text_stream:
            full_response += text
            chunk_count += 1

            # 100チャンクごとに進捗を表示
            if chunk_count % 100 == 0:
                print(f"[{chunk_count} chunks received...]")

    final_message = stream.get_final_message()
    usage = final_message.usage

    return {
        "response": full_response,
        "chunks": chunk_count,
        "input_tokens": usage.input_tokens,
        "output_tokens": usage.output_tokens,
        "total_tokens": usage.input_tokens + usage.output_tokens
    }

# 使用例
result = stream_and_collect([
    {
        "role": "user",
        "content": "科学的手法の詳細な解説をしてください"
    }
])

print(f"Response length: {len(result['response'])} characters")
print(f"Chunks: {result['chunks']}")
print(f"Tokens used: {result['total_tokens']}")
```

## マルチターン会話の実装

### チャットセッションの基本構造

```python
from anthropic import Anthropic

client = Anthropic()

class ChatSession:
    def __init__(self, system_prompt=None):
        self.conversation_history = []
        self.system_prompt = system_prompt or "You are a helpful AI assistant."
        self.model = "gemini-2.5-pro"

    def add_message(self, role, content):
        """会話履歴にメッセージを追加"""
        self.conversation_history.append({
            "role": role,
            "content": content
        })

    def send_message(self, user_message):
        """ユーザーメッセージを送信し、応答を得る"""
        # ユーザーメッセージを履歴に追加
        self.add_message("user", user_message)

        # APIに現在の履歴全体を送信
        response = client.messages.create(
            model=self.model,
            max_tokens=1024,
            system=self.system_prompt,
            messages=self.conversation_history
        )

        # アシスタントの応答を抽出
        assistant_message = response.content[0].text

        # 応答を履歴に追加
        self.add_message("assistant", assistant_message)

        return assistant_message

    def get_history(self):
        """会話履歴を取得"""
        return self.conversation_history

    def clear_history(self):
        """会話履歴をクリア"""
        self.conversation_history = []

# 使用例
chat = ChatSession(
    system_prompt="あなたはPythonのエキスパートです。技術的な質問に対して詳しく説明します。"
)

# ターン1
response1 = chat.send_message("リスト内包表記について説明してください")
print(f"Q: リスト内包表記について説明してください\nA: {response1}\n")

# ターン2
response2 = chat.send_message("その例を5つ示してください")
print(f"Q: その例を5つ示してください\nA: {response2}\n")

# ターン3
response3 = chat.send_message("パフォーマンス面での利点は？")
print(f"Q: パフォーマンス面での利点は？\nA: {response3}\n")
```

### ストリーミングとマルチターンの組み合わせ

```python
class StreamingChatSession(ChatSession):
    """ストリーミング対応のチャットセッション"""

    def send_message_streaming(self, user_message):
        """ストリーミング応答でメッセージを送信"""
        # ユーザーメッセージを履歴に追加
        self.add_message("user", user_message)

        full_response = ""

        # ストリーミング実行
        with client.messages.stream(
            model=self.model,
            max_tokens=1024,
            system=self.system_prompt,
            messages=self.conversation_history
        ) as stream:
            for text in stream.text_stream:
                full_response += text
                print(text, end="", flush=True)

        print()  # 改行

        # 完全な応答を履歴に追加
        self.add_message("assistant", full_response)

        return full_response

# 使用例
streaming_chat = StreamingChatSession(
    system_prompt="あなたはデータサイエンスの専門家です。詳しく、わかりやすく説明します。"
)

print("Question 1:")
streaming_chat.send_message_streaming("機械学習とディープラーニングの違いは？")

print("\nQuestion 2:")
streaming_chat.send_message_streaming("ニューラルネットワークについてもっと詳しく")
```

## 会話コンテキストの管理

### トークン数を考慮した履歴管理

長い会話ではトークン数が増加し、APIコストが上昇します：

```python
class ContextAwareChatSession(ChatSession):
    def __init__(self, system_prompt=None, max_tokens_per_request=4000):
        super().__init__(system_prompt)
        self.max_tokens_per_request = max_tokens_per_request

    def estimate_tokens(self, text):
        """テキストのトークン数を簡易推定（1単語≒1.3トークン）"""
        return int(len(text.split()) * 1.3)

    def trim_history(self):
        """会話履歴がトークン数を超えないようにトリミング"""
        estimated_tokens = self.estimate_tokens(
            str(self.conversation_history)
        )

        # 推定トークン数がリミットを超えた場合、古いメッセージから削除
        while estimated_tokens > self.max_tokens_per_request and len(self.conversation_history) > 1:
            self.conversation_history.pop(0)
            estimated_tokens = self.estimate_tokens(
                str(self.conversation_history)
            )

        return len(self.conversation_history)

    def send_message(self, user_message):
        """会話履歴を最適化してから送信"""
        self.add_message("user", user_message)

        # 必要に応じて古いメッセージを削除
        remaining = self.trim_history()
        print(f"[Conversation context: {remaining} messages]")

        response = client.messages.create(
            model=self.model,
            max_tokens=1024,
            system=self.system_prompt,
            messages=self.conversation_history
        )

        assistant_message = response.content[0].text
        self.add_message("assistant", assistant_message)

        return assistant_message

# 使用例
context_aware_chat = ContextAwareChatSession(
    system_prompt="Helpful assistant",
    max_tokens_per_request=3000
)

# 長い会話でも自動的にコンテキストが管理される
for i in range(10):
    response = context_aware_chat.send_message(f"質問 {i+1}: これについて説明してください")
```

## 実践的なアプリケーション例

### インタラクティブなコーディングアシスタント

```python
def coding_assistant():
    """Pythonコーディングアシスタント"""
    chat = StreamingChatSession(
        system_prompt="""
あなたはシニアPythonデベロッパーです。
- クリーンで効率的なコード例を提供する
- ベストプラクティスを説明する
- パフォーマンスと可読性のバランスを重視する
"""
    )

    print("=== Python Coding Assistant ===")
    print("Type 'quit' to exit\n")

    while True:
        user_input = input("You: ").strip()
        if user_input.lower() == 'quit':
            break

        if user_input:
            print("Assistant: ", end="")
            chat.send_message_streaming(user_input)
            print()

# 実行
if __name__ == "__main__":
    coding_assistant()
```

## ベストプラクティス

1. **適切なストリーミング活用**: UI側のリアルタイム表示が必要な場合のみストリーミングを使用
2. **システムプロンプトの工夫**: AIの振る舞いを制御する最も効果的な方法
3. **コンテキスト管理**: 長い会話ではトークン数を監視し、必要に応じて古いメッセージを削除
4. **エラーハンドリング**: ネットワーク接続が不安定な環境ではストリーミングが中断される可能性を考慮
5. **費用管理**: 不要な再生成や過度なAPIコールを避ける

Gemini APIのストリーミングとマルチターン会話機能を活用することで、リアルタイムで応答性の高いAIアプリケーションを構築できます。
