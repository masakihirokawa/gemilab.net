## はじめに

**Veo 3** は、Google DeepMindが開発した最先端の動画生成AI モデルです。テキストプロンプトまたは静止画から、高品質な動画を自動生成できます。

### Veo 3 の特徴

- **解像度**: 最大1080p（フルHD）での生成に対応
- **長さ**: 最大8秒のビデオクリップを生成可能
- **マルチモーダル**: テキスト入力、画像からのアニメーション生成に対応
- **オーディオ対応**: 生成動画と別途、音声の埋め込みが可能
- **Google AI スタック統合**: Gemini APIを通じて利用可能

Google AI エコシステムにおいて、Veo 3 はコンテンツ生成の新しい次元を開拓します。Gemini 2.0 Flash以上のモデルと組み合わせることで、テキスト分析→動画生成→メタデータ抽出という統合ワークフロー構築が実現します。

<Callout type="info">
本ガイドは、Veo 3をプロダクション環境で運用するエンジニア向けの実装パターンを網羅しています。認証設定からエラーハンドリング、コスト最適化まで、実務で即座に活用できるコード例を提供します。
</Callout>

## テキスト→動画生成（Text-to-Video）

テキストプロンプトから動画を生成する最も一般的なユースケースです。Veo 3 は詳細なシーン説明、カメラワーク、スタイル指定などを支持しており、柔軟な動画制作が可能です。

### 基本的な実装例

```python
import google.generativeai as genai
import os
import time
from pathlib import Path

def generate_video_from_text(prompt: str, output_path: str = "output.mp4"):
    """
    テキストプロンプトから動画を生成

    Args:
        prompt: 動画生成用プロンプト
        output_path: 生成動画の保存先

    Returns:
        生成された動画のパス
    """

    client = genai.Client(api_key=os.getenv("GOOGLE_API_KEY"))

    print(f"動画生成を開始します: {prompt}")

    # 動画生成リクエストを送信（非同期処理）
    response = client.models.generate_video(
        model="models/veo-3",
        prompt=prompt,
        config={
            "duration_seconds": 8,
            "resolution": "1080p",
            "frame_rate": 30
        }
    )

    # 生成完了までポーリング（最大15分）
    start_time = time.time()
    max_wait_time = 15 * 60  # 15分

    while response.state == "PROCESSING":
        elapsed = time.time() - start_time
        if elapsed > max_wait_time:
            raise TimeoutError("動画生成がタイムアウトしました")

        print(f"生成中... ({elapsed:.0f}秒経過)")
        time.sleep(10)  # 10秒ごとにポーリング
        response = client.models.get_generate_video_response(response.id)

    if response.state == "FAILED":
        raise RuntimeError(f"動画生成エラー: {response.error_message}")

    # 動画をファイルに保存
    video_data = response.video_bytes

    with open(output_path, "wb") as f:
        f.write(video_data)

    print(f"動画が保存されました: {output_path}")
    return output_path


# 実行例
if __name__ == "__main__":
    prompt = """
    Cinematic wide shot of a serene Japanese garden at sunrise.
    A stone lantern sits peacefully near a still pond, reflecting
    the golden morning light. Cherry blossoms gently fall from above.
    Slow dolly move forward, gentle 24fps cinematic motion.
    Color grade: warm, soft focus edges, cinematic depth of field.
    """

    generate_video_from_text(prompt, "japan_garden.mp4")
```

## 画像→動画生成（Image-to-Video）

既存の静止画像をアニメーション化して動画に変換できます。これはプロダクト写真、イラスト、建築画像などの活性化に有用です。

### 実装例

```python
import google.generativeai as genai
from pathlib import Path
import base64
import time
import os

def animate_image_to_video(
    image_path: str,
    motion_prompt: str,
    output_path: str = "animated.mp4"
):
    """
    静止画像を動画にアニメーション化

    Args:
        image_path: 入力画像ファイルパス
        motion_prompt: 動きに関する指示プロンプト
        output_path: 出力動画ファイルパス

    Returns:
        生成動画のパス
    """

    client = genai.Client(api_key=os.getenv("GOOGLE_API_KEY"))

    # 画像をBase64エンコード
    image_data = Path(image_path).read_bytes()
    image_b64 = base64.b64encode(image_data).decode('utf-8')

    # ファイル拡張子からMIMEタイプを推定
    extension = Path(image_path).suffix.lower()
    mime_types = {
        ".jpg": "image/jpeg",
        ".jpeg": "image/jpeg",
        ".png": "image/png",
        ".gif": "image/gif",
        ".webp": "image/webp"
    }
    mime_type = mime_types.get(extension, "image/jpeg")

    print(f"画像をアップロード: {image_path}")

    # 画像アニメーション生成リクエスト
    response = client.models.generate_video(
        model="models/veo-3",
        image_source=image_b64,
        mime_type=mime_type,
        prompt=motion_prompt,
        config={
            "duration_seconds": 4,
            "resolution": "1080p",
            "animation_style": "smooth",
            "frame_rate": 24
        }
    )

    # ポーリングループ
    start_time = time.time()
    max_wait = 20 * 60  # 20分

    while response.state == "PROCESSING":
        elapsed = time.time() - start_time
        if elapsed > max_wait:
            raise TimeoutError("画像アニメーション生成がタイムアウト")

        percentage = response.progress_percent
        print(f"生成中: {percentage}% ({elapsed:.0f}秒経過)")
        time.sleep(5)
        response = client.models.get_generate_video_response(response.id)

    if response.state == "FAILED":
        raise RuntimeError(f"生成失敗: {response.error_message}")

    # ファイル保存
    with open(output_path, "wb") as f:
        f.write(response.video_bytes)

    print(f"アニメーション動画: {output_path}")
    return output_path


# 実行例
if __name__ == "__main__":
    motion_prompt = """
    The landscape comes alive with subtle motion.
    Wind gently moves the grass and trees.
    Clouds slowly drift across the sky.
    Camera perspective remains static.
    Preserve original composition and colors.
    """

    animate_image_to_video(
        "landscape.png",
        motion_prompt,
        "animated_landscape.mp4"
    )
```

<Callout type="info">
画像→動画生成では、元の画像の構図やカラーパレットが保持されるため、既存のビジュアルアイデンティティを維持しながら動画化できます。
</Callout>

## プロンプトエンジニアリング

Veo 3 の出力品質は、プロンプトの質に大きく左右されます。プロダクション環境では、プロンプトテンプレートとセクション化が推奨されます。

### カメラワーク用語

```
パン（Pan）: カメラが左右に回転
ティルト（Tilt）: カメラが上下に回転
ドリー（Dolly）: カメラが前後に移動
ズーム（Zoom）: 焦点距離の変更
トラック（Track）: サイドに移動
ロール（Roll）: カメラが時計回りに回転
```

### テンプレートベースのプロンプト生成

```python
class VideoPromptBuilder:
    """プロダクションレディーなプロンプト生成"""

    def __init__(self):
        self.sections = {}

    def set_scene(self, scene_description: str):
        """シーンの設定"""
        self.sections['scene'] = scene_description
        return self

    def set_camera_motion(self, camera_work: str):
        """カメラワーク"""
        self.sections['camera'] = camera_work
        return self

    def set_lighting(self, lighting_style: str):
        """照明スタイル"""
        self.sections['lighting'] = lighting_style
        return self

    def set_style(self, visual_style: str):
        """映像スタイル"""
        self.sections['style'] = visual_style
        return self

    def set_duration(self, seconds: int):
        """動画長（秒）"""
        self.sections['duration'] = f"{seconds} seconds"
        return self

    def add_negative_prompt(self, negative: str):
        """避けるべき要素"""
        self.sections['negative'] = negative
        return self

    def build(self) -> str:
        """最終プロンプトを生成"""
        parts = []

        if 'scene' in self.sections:
            parts.append(f"Scene: {self.sections['scene']}")
        if 'camera' in self.sections:
            parts.append(f"Camera: {self.sections['camera']}")
        if 'lighting' in self.sections:
            parts.append(f"Lighting: {self.sections['lighting']}")
        if 'style' in self.sections:
            parts.append(f"Style: {self.sections['style']}")
        if 'duration' in self.sections:
            parts.append(f"Duration: {self.sections['duration']}")
        if 'negative' in self.sections:
            parts.append(f"NOT: {self.sections['negative']}")

        return "\n".join(parts)


# 使用例
prompt = (VideoPromptBuilder()
    .set_scene("Futuristic cyberpunk city street at night, neon signs glowing")
    .set_camera_motion("Slow forward dolly through the street, pan to show storefronts")
    .set_lighting("Neon pink and cyan ambient lighting, rain-wet surfaces")
    .set_style("Cinematic cyberpunk, photorealistic, film noir color grade")
    .set_duration(8)
    .add_negative_prompt("cartoon, blurry, low quality, watermarks")
    .build())

print(prompt)
```

## 非同期処理とバッチ生成

プロダクション環境では、複数の動画を効率的に処理する非同期パターンが必須です。

### 非同期ポーリングの実装

```python
import asyncio
import google.generativeai as genai
from typing import List
import os
from dataclasses import dataclass
from datetime import datetime

@dataclass
class GenerationJob:
    """動画生成ジョブの追跡"""
    job_id: str
    prompt: str
    output_path: str
    created_at: datetime
    status: str = "PENDING"
    progress: int = 0
    error: str = None

class VideoGenerationQueue:
    """非同期動画生成キュー"""

    def __init__(self, api_key: str, max_concurrent: int = 5):
        self.client = genai.Client(api_key=api_key)
        self.max_concurrent = max_concurrent
        self.jobs: dict[str, GenerationJob] = {}
        self.semaphore = asyncio.Semaphore(max_concurrent)

    async def submit_job(
        self,
        prompt: str,
        output_path: str
    ) -> str:
        """生成ジョブを送信"""
        response = self.client.models.generate_video(
            model="models/veo-3",
            prompt=prompt,
            config={
                "duration_seconds": 8,
                "resolution": "1080p"
            }
        )

        job = GenerationJob(
            job_id=response.id,
            prompt=prompt,
            output_path=output_path,
            created_at=datetime.now(),
            status="PROCESSING"
        )

        self.jobs[response.id] = job

        asyncio.create_task(
            self._poll_job(response.id)
        )

        return response.id

    async def _poll_job(self, job_id: str):
        """ジョブのポーリング"""
        async with self.semaphore:
            job = self.jobs[job_id]
            max_wait = 20 * 60  # 20分
            elapsed = 0
            poll_interval = 10  # 10秒

            while elapsed < max_wait:
                try:
                    response = self.client.models.get_generate_video_response(
                        job_id
                    )

                    job.progress = response.progress_percent

                    if response.state == "COMPLETED":
                        with open(job.output_path, "wb") as f:
                            f.write(response.video_bytes)
                        job.status = "COMPLETED"
                        print(f"✓ 完了: {job_id}")
                        return

                    elif response.state == "FAILED":
                        job.status = "FAILED"
                        job.error = response.error_message
                        print(f"✗ 失敗: {job_id} - {response.error_message}")
                        return

                    await asyncio.sleep(poll_interval)
                    elapsed += poll_interval

                except Exception as e:
                    job.status = "ERROR"
                    job.error = str(e)
                    print(f"エラー: {job_id} - {e}")
                    return

            job.status = "TIMEOUT"
            job.error = "生成がタイムアウト"

    async def wait_for_completion(
        self,
        job_ids: List[str],
        timeout_seconds: int = 1200
    ) -> dict[str, GenerationJob]:
        """複数ジョブの完了を待機"""
        start = datetime.now()

        while True:
            elapsed = (datetime.now() - start).total_seconds()
            if elapsed > timeout_seconds:
                break

            all_done = all(
                self.jobs[jid].status in ["COMPLETED", "FAILED", "TIMEOUT"]
                for jid in job_ids
            )

            if all_done:
                break

            await asyncio.sleep(5)

        return {
            jid: self.jobs[jid]
            for jid in job_ids
        }


# 実行例
async def batch_generate_videos():
    """複数の動画をバッチ生成"""
    queue = VideoGenerationQueue(
        api_key=os.getenv("GOOGLE_API_KEY"),
        max_concurrent=3
    )

    prompts = [
        ("Sunset over ocean waves", "sunset.mp4"),
        ("Futuristic city skyline", "city.mp4"),
        ("Forest canopy with sunlight", "forest.mp4"),
    ]

    job_ids = []
    for prompt, output in prompts:
        job_id = await queue.submit_job(prompt, output)
        job_ids.append(job_id)

    # 全ジョブの完了を待機
    results = await queue.wait_for_completion(job_ids, timeout_seconds=1800)

    for job_id, job in results.items():
        print(f"{job_id}: {job.status} - {job.output_path}")


if __name__ == "__main__":
    asyncio.run(batch_generate_videos())
```

## TypeScript/Next.js での実装

### API ルートの実装

```typescript
import { GoogleGenerativeAI } from "@google/generative-ai";
import { NextRequest, NextResponse } from "next/server";
import fs from "fs/promises";
import path from "path";

const client = new GoogleGenerativeAI(
  process.env.GOOGLE_API_KEY || ""
);

export async function POST(request: NextRequest) {
  try {
    const body = await request.json();
    const { prompt, outputFilename } = body;

    if (!prompt) {
      return NextResponse.json(
        { error: "prompt is required" },
        { status: 400 }
      );
    }

    console.log("Starting video generation:", prompt);

    // 動画生成リクエスト送信
    const response = await client.models
      .generateVideo({
        model: "models/veo-3",
        prompt: prompt,
        config: {
          duration_seconds: 8,
          resolution: "1080p",
          frame_rate: 30,
        },
      });

    let currentResponse = response;

    // ポーリングループ
    const maxWaitTime = 15 * 60 * 1000; // 15分
    const startTime = Date.now();

    while (currentResponse.state === "PROCESSING") {
      const elapsed = Date.now() - startTime;

      if (elapsed > maxWaitTime) {
        return NextResponse.json(
          { error: "Video generation timeout" },
          { status: 408 }
        );
      }

      console.log(
        `Progress: ${currentResponse.progress_percent}% (${elapsed / 1000}s)`
      );

      await new Promise((resolve) => setTimeout(resolve, 10000)); // 10秒待機

      currentResponse = await client.models.getGenerateVideoResponse(
        currentResponse.id
      );
    }

    if (currentResponse.state === "FAILED") {
      return NextResponse.json(
        { error: `Generation failed: ${currentResponse.error_message}` },
        { status: 500 }
      );
    }

    // 動画をファイルに保存
    const videosDir = path.join(process.cwd(), "public", "videos");
    await fs.mkdir(videosDir, { recursive: true });

    const filename =
      outputFilename ||
      `video_${Date.now()}.mp4`;
    const filepath = path.join(videosDir, filename);

    if (currentResponse.video_bytes) {
      const buffer = Buffer.from(currentResponse.video_bytes);
      await fs.writeFile(filepath, buffer);
    }

    return NextResponse.json(
      {
        success: true,
        videoPath: `/videos/${filename}`,
        message: "Video generated successfully",
      },
      { status: 200 }
    );
  } catch (error) {
    console.error("Error:", error);
    return NextResponse.json(
      {
        error: error instanceof Error ? error.message : "Unknown error",
      },
      { status: 500 }
    );
  }
}
```

#### 2. キャッシングと再利用

```python
import hashlib
import json

class VideoCache:
    """生成済み動画のキャッシング"""

    def __init__(self, cache_dir: str = ".cache/videos"):
        self.cache_dir = cache_dir

    def get_cache_key(self, prompt: str, config: dict) -> str:
        """プロンプトと設定からキャッシュキーを生成"""
        cache_data = json.dumps({
            "prompt": prompt,
            "config": config
        }, sort_keys=True)

        return hashlib.sha256(cache_data.encode()).hexdigest()

    def get_cached_video(self, prompt: str, config: dict) -> str | None:
        """キャッシュから動画を取得"""
        key = self.get_cache_key(prompt, config)
        cache_file = f"{self.cache_dir}/{key}.mp4"

        import os
        if os.path.exists(cache_file):
```

```python
def batch_generate_with_cost_tracking(
```

### 一般的なエラーと解決策

```python
from enum import Enum
from typing import Optional

class GenerationError(Enum):
    """動画生成エラータイプ"""
    INVALID_API_KEY = "API キーが無効です"
    QUOTA_EXCEEDED = "クォータ超過（月間制限に達しました）"
    INVALID_PROMPT = "プロンプトが無効または不適切です"
    TIMEOUT = "生成がタイムアウトしました"
    NETWORK_ERROR = "ネットワークエラー"
    UNSUPPORTED_FORMAT = "サポートされていない画像形式"

def handle_generation_error(error: Exception) -> Optional[GenerationError]:
    """エラーハンドリングと診断"""
    error_str = str(error).lower()

    if "authentication" in error_str or "unauthorized" in error_str:
        return GenerationError.INVALID_API_KEY
    elif "quota" in error_str or "limit" in error_str:
        return GenerationError.QUOTA_EXCEEDED
    elif "invalid prompt" in error_str:
        return GenerationError.INVALID_PROMPT
    elif "timeout" in error_str:
        return GenerationError.TIMEOUT
    elif "connection" in error_str:
        return GenerationError.NETWORK_ERROR
    elif "format" in error_str:
        return GenerationError.UNSUPPORTED_FORMAT

    return None
```

```python
import logging
from datetime import datetime

class VideoGenerationLogger:
    """動画生成のロギング"""

    def __init__(self):
        self.logger = logging.getLogger("VideoGeneration")
        handler = logging.FileHandler(f"logs/veo3_{datetime.now().isoformat()}.log")
        formatter = logging.Formatter(
            '%(asctime)s - %(name)s - %(levelname)s - %(message)s'
        )
        handler.setFormatter(formatter)
        self.logger.addHandler(handler)
        self.logger.setLevel(logging.INFO)

    def log_generation_start(self, prompt: str, config: dict):
        """生成開始をログ"""
        self.logger.info(f"Generation started | Prompt: {prompt[:50]}... | Config: {config}")

    def log_progress(self, job_id: str, progress: int, elapsed_seconds: float):
        """進捗をログ"""
        self.logger.info(f"Job {job_id} | Progress: {progress}% | Elapsed: {elapsed_seconds}s")

    def log_completion(self, job_id: str, output_path: str, total_seconds: float):
        """完了をログ"""
        self.logger.info(f"Job {job_id} completed | Output: {output_path} | Duration: {total_seconds}s")

    def log_error(self, job_id: str, error: str):
        """エラーをログ"""
        self.logger.error(f"Job {job_id} failed | Error: {error}")
```
