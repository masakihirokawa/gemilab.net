# Gemini リアルタイムストリーミング高度パターン — SSE・WebSocket連携

Gemini API のストリーミング機能を活用することで、遅延を最小化した高速なリアルタイムアプリケーションを構築できます。このガイドでは、バックエンド・フロントエンドの統合実装パターンを解説します。

## ストリーミングの基本

### Python での基本的なストリーミング

```python
from anthropic import Anthropic

client = Anthropic()

def stream_basic():
    """基本的なストリーミング例"""
    with client.messages.stream(
        model="gemini-2.5-flash",
        max_tokens=1024,
        messages=[{
            "role": "user",
            "content": "Python の非同期プログラミングについて説明してください"
        }]
    ) as stream:
        for text in stream.text_stream:
            print(text, end="", flush=True)
    print()

# 実行
stream_basic()
```

<Callout type="info">
ストリーミングにより、全レスポンスの完成を待たずにトークンを順序どおりに受け取ることができます。これにより、UX の体感速度が大幅に向上します。
</Callout>

## Server-Sent Events (SSE) エンドポイント

### FastAPI での SSE 実装

```python
from fastapi import FastAPI, Response
from fastapi.responses import StreamingResponse
from anthropic import Anthropic
import asyncio

app = FastAPI()
client = Anthropic()

@app.post("/api/stream")
async def stream_response(request_data: dict):
    """SSE エンドポイント"""

    async def generate():
        """ストリーミングジェネレータ"""
        user_message = request_data.get("message", "")

        try:
            # Gemini API のストリーミング開始
            with client.messages.stream(
                model="gemini-2.5-flash",
                max_tokens=2048,
                messages=[{
                    "role": "user",
                    "content": user_message
                }]
            ) as stream:
                for text in stream.text_stream:
                    # SSE フォーマットで送信
                    yield f"data: {text}\n\n"

        except Exception as e:
            yield f"data: ERROR: {str(e)}\n\n"

    return StreamingResponse(
        generate(),
        media_type="text/event-stream",
        headers={
            "Cache-Control": "no-cache",
            "Connection": "keep-alive",
            "X-Accel-Buffering": "no"
        }
    )
```

## WebSocket 統合

### FastAPI での WebSocket 実装

```python
from fastapi import FastAPI, WebSocket
from fastapi.websockets import WebSocketDisconnect
from anthropic import Anthropic
import asyncio
import json

app = FastAPI()
client = Anthropic()

@app.websocket("/ws/chat")
async def websocket_endpoint(websocket: WebSocket):
    """WebSocket チャットエンドポイント"""
    await websocket.accept()

    try:
        while True:
            # クライアントからメッセージ受信
            data = await websocket.receive_text()
            message = json.loads(data)

            # Gemini ストリーミング処理
            response_text = ""
            try:
                with client.messages.stream(
                    model="gemini-2.5-flash",
                    max_tokens=2048,
                    messages=[{
                        "role": "user",
                        "content": message.get("content", "")
                    }]
                ) as stream:
                    for text in stream.text_stream:
                        response_text += text
                        # リアルタイムでクライアントに送信
                        await websocket.send_json({
                            "type": "stream",
                            "content": text
                        })

                # ストリーミング完了
                await websocket.send_json({
                    "type": "complete",
                    "total_tokens": len(response_text.split())
                })

            except Exception as e:
                await websocket.send_json({
                    "type": "error",
                    "message": str(e)
                })

    except WebSocketDisconnect:
        print("クライアント接続切断")
```

<Callout type="warning">
WebSocket 接続はサーバーリソースを消費します。接続数制限、タイムアウト、クリーンアップを適切に実装してください。
</Callout>

## チャンク処理と前処理

### チャンク単位での処理

```python
import re
from typing import Generator

class StreamChunkProcessor:
    """ストリームチャンクを処理"""

    def __init__(self):
        self.buffer = ""
        self.sentence_pattern = re.compile(r'[。！？\n]')

    def process_chunk(self, chunk: str) -> Generator[str, None, None]:
        """チャンクを処理し、完成した文を返す"""
        self.buffer += chunk

        # 句読点で分割
        sentences = self.sentence_pattern.split(self.buffer)

        # 最後の不完全な文を除外
        for sentence in sentences[:-1]:
            if sentence.strip():
                yield sentence.strip() + "\n"

        # 未完成の文を保持
        self.buffer = sentences[-1] if sentences[-1] else ""

    def flush(self) -> str:
        """バッファの残りを返す"""
        result = self.buffer
        self.buffer = ""
        return result

# 使用例
processor = StreamChunkProcessor()

with client.messages.stream(
    model="gemini-2.5-flash",
    max_tokens=1024,
    messages=[{"role": "user", "content": "物語を書いてください"}]
) as stream:
    for chunk in stream.text_stream:
        for sentence in processor.process_chunk(chunk):
            print(f"完成した文: {sentence}")

    # 残りを処理
    final = processor.flush()
    if final:
        print(f"最終部分: {final}")
```

## バックプレッシャー制御

### キュー形式でのバッファリング

```python
import asyncio
import queue
from typing import Optional

class StreamBuffer:
    """バックプレッシャー制御付きストリームバッファ"""

    def __init__(self, max_size: int = 100):
        self.queue = asyncio.Queue(maxsize=max_size)
        self.done = False

    async def put(self, item: str):
        """アイテムをキューに追加（バックプレッシャー制御）"""
        try:
            await asyncio.wait_for(
                self.queue.put(item),
                timeout=5.0
            )
        except asyncio.TimeoutError:
            print("バッファオーバーフロー: 生産速度が高すぎます")

    async def get(self) -> Optional[str]:
        """キューからアイテムを取得"""
        if self.done and self.queue.empty():
            return None
        try:
            return await asyncio.wait_for(
                self.queue.get(),
                timeout=5.0
            )
        except asyncio.TimeoutError:
            return None

    def set_done(self):
        """ストリーミング完了を標示"""
        self.done = True

# 使用例
async def stream_with_backpressure():
    """バックプレッシャー制御付きストリーミング"""
    buffer = StreamBuffer(max_size=50)

    async def producer():
        """プロデューサータスク"""
        with client.messages.stream(
            model="gemini-2.5-flash",
            max_tokens=2048,
            messages=[{"role": "user", "content": "長いテキストを生成してください"}]
        ) as stream:
            for chunk in stream.text_stream:
                await buffer.put(chunk)
        buffer.set_done()

    async def consumer():
        """コンシューマータスク"""
        while True:
            item = await buffer.get()
            if item is None:
                break
            print(f"処理: {item}", end="", flush=True)

    # 並行実行
    await asyncio.gather(
        producer(),
        consumer()
    )

# 実行
asyncio.run(stream_with_backpressure())
```

## フロントエンド統合（TypeScript/JavaScript）

### SSE コンシューマー

```typescript
// Frontend SSE Consumer
class GeminiSSEClient {
  private baseUrl: string = "http://localhost:8000";

  async streamMessage(userMessage: string, onChunk: (text: string) => void) {
    try {
      const response = await fetch(`${this.baseUrl}/api/stream`, {
        method: "POST",
        headers: {
          "Content-Type": "application/json",
        },
        body: JSON.stringify({ message: userMessage }),
      });

      if (!response.ok) {
        throw new Error(`HTTP error! status: ${response.status}`);
      }

      const reader = response.body?.getReader();
      if (!reader) throw new Error("No response body");

      const decoder = new TextDecoder();
      let buffer = "";

      while (true) {
        const { done, value } = await reader.read();
        if (done) break;

        buffer += decoder.decode(value, { stream: true });

        // SSE パースロジック
        const lines = buffer.split("\n");
        buffer = lines.pop() || ""; // 不完全な行をバッファに保持

        for (const line of lines) {
          if (line.startsWith("data: ")) {
            const data = line.slice(6); // "data: " を除去
            if (data !== "[DONE]") {
              onChunk(data);
            }
          }
        }
      }
    } catch (error) {
      console.error("Streaming error:", error);
    }
  }
}

// 使用例
const client = new GeminiSSEClient();
const outputElement = document.getElementById("output");

document.getElementById("sendBtn").addEventListener("click", async () => {
  const input = (document.getElementById("input") as HTMLInputElement).value;
  outputElement.textContent = ""; // クリア

  await client.streamMessage(input, (chunk) => {
    outputElement.textContent += chunk;
    outputElement.scrollTop = outputElement.scrollHeight;
  });
});
```

### WebSocket コンシューマー

```typescript
// Frontend WebSocket Consumer
class GeminiWebSocketClient {
  private ws: WebSocket | null = null;
  private url: string = "ws://localhost:8000/ws/chat";

  connect(onMessage: (data: any) => void) {
    this.ws = new WebSocket(this.url);

    this.ws.onopen = () => {
      console.log("WebSocket connected");
    };

    this.ws.onmessage = (event) => {
      const data = JSON.parse(event.data);
      onMessage(data);
    };

    this.ws.onerror = (error) => {
      console.error("WebSocket error:", error);
    };

    this.ws.onclose = () => {
      console.log("WebSocket disconnected");
    };
  }

  send(message: string) {
    if (this.ws && this.ws.readyState === WebSocket.OPEN) {
      this.ws.send(
        JSON.stringify({
          content: message,
          timestamp: new Date().toISOString(),
        })
      );
    }
  }

  disconnect() {
    if (this.ws) {
      this.ws.close();
    }
  }
}

// 使用例
const wsClient = new GeminiWebSocketClient();
const outputElement = document.getElementById("output");

wsClient.connect((data) => {
  if (data.type === "stream") {
    outputElement.textContent += data.content;
  } else if (data.type === "complete") {
    outputElement.textContent += "\n\n[完了]";
  } else if (data.type === "error") {
    outputElement.textContent += `\n[エラー] ${data.message}`;
  }
});

document.getElementById("sendBtn").addEventListener("click", () => {
  const input = (document.getElementById("input") as HTMLInputElement).value;
  outputElement.textContent = "";
  wsClient.send(input);
  (document.getElementById("input") as HTMLInputElement).value = "";
});

window.addEventListener("beforeunload", () => {
  wsClient.disconnect();
});
```

## レイテンシー最適化

### キャッシング戦略

```python
from functools import lru_cache
import hashlib

class StreamingCache:
    """ストリーミングレスポンスのキャッシング"""

    def __init__(self):
        self.cache = {}

    def get_cache_key(self, prompt: str) -> str:
        """プロンプトのハッシュをキーとして使用"""
        return hashlib.md5(prompt.encode()).hexdigest()

    def stream_with_cache(self, model: str, prompt: str, max_age_hours: int = 24):
        """キャッシュ付きストリーミング"""
        cache_key = self.get_cache_key(prompt)

        # キャッシュ確認
        if cache_key in self.cache:
            cached_response, cached_time = self.cache[cache_key]
            age_hours = (datetime.now() - cached_time).total_seconds() / 3600
            if age_hours < max_age_hours:
                # キャッシュから返す
                for chunk in cached_response.split("\n"):
                    if chunk:
                        yield chunk
                return

        # API へのリクエスト
        response_parts = []
        with client.messages.stream(
            model=model,
            max_tokens=2048,
            messages=[{"role": "user", "content": prompt}]
        ) as stream:
            for text in stream.text_stream:
                response_parts.append(text)
                yield text

        # キャッシュに保存
        full_response = "".join(response_parts)
        self.cache[cache_key] = (full_response, datetime.now())

# 使用例
from datetime import datetime

cache = StreamingCache()
for chunk in cache.stream_with_cache("gemini-2.5-flash", "Python について教えてください"):
    print(chunk, end="", flush=True)
```

<Callout type="info">
キャッシング戦略により、繰り返しのリクエストにおいて、API 呼び出しを削減し、レスポンス速度を大幅に向上させます。
</Callout>

## エラーハンドリングと回復

### リトライとフォールバック

```python
from tenacity import retry, stop_after_attempt, wait_exponential
import logging

logger = logging.getLogger(__name__)

class RobustStreamingClient:
    """堅牢なストリーミングクライアント"""

    def __init__(self, client, max_retries: int = 3):
        self.client = client
        self.max_retries = max_retries

    @retry(
        stop=stop_after_attempt(3),
        wait=wait_exponential(multiplier=1, min=2, max=10)
    )
    def stream_with_retry(self, model: str, prompt: str):
        """リトライ付きストリーミング"""
        try:
            with self.client.messages.stream(
                model=model,
                max_tokens=2048,
                messages=[{"role": "user", "content": prompt}]
            ) as stream:
                for text in stream.text_stream:
                    yield text
        except Exception as e:
            logger.error(f"Streaming error: {str(e)}")
            raise

    async def stream_with_fallback(self, prompt: str):
        """フォールバック機能付きストリーミング"""
        models = [
            "gemini-2.5-pro",
            "gemini-2.5-flash",
            "gemini-2.5-flash-lite"
        ]

        for model in models:
            try:
                logger.info(f"Trying model: {model}")
                async for chunk in self.stream_with_retry(model, prompt):
                    yield chunk
                return
            except Exception as e:
                logger.warning(f"Model {model} failed: {str(e)}")
                continue

        raise Exception("All models failed")

# 使用例
robust_client = RobustStreamingClient(client)
for chunk in robust_client.stream_with_retry("gemini-2.5-flash", "テストプロンプト"):
    print(chunk, end="", flush=True)
```
