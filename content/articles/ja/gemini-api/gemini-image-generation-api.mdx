## Gemini 画像生成APIの基本

Gemini 画像生成APIは、テキスト説明から高品質な画像を生成するための強力なツールです。Google傘下のImagen技術を統合しており、以下のような用途に対応します：

- マーケティング資料やソーシャルメディアコンテンツ
- アート制作やデザイン補助
- 製品プロトタイプのビジュアル化
- 教育やトレーニング用の図解生成
- ゲーム開発での背景やキャラクター設計

このガイドでは、API利用の基本から高度なカスタマイズまでを網羅します。

## API認証とセットアップ

### 初期設定

```python
import google.generativeai as genai
import os
from pathlib import Path

# APIキーの設定
api_key = os.getenv("GEMINI_API_KEY")
genai.configure(api_key=api_key)

def setup_image_generation():
    """画像生成APIのセットアップ"""
    # 利用可能なモデルを確認
    models = genai.list_models()
    image_models = [m for m in models if 'imagen' in m.name.lower()]

    if image_models:
        print(f"利用可能な画像生成モデル: {image_models[0].name}")
        return image_models[0].name
    else:
        # デフォルトモデルを返す
        return "gemini-2.0-flash"

model_name = setup_image_generation()
print(f"使用モデル: {model_name}")

# 出力例:
# 利用可能な画像生成モデル: gemini-2.0-flash
# 使用モデル: gemini-2.0-flash
```

## シンプルな画像生成

### 基本的なテキストから画像への変換

```python
from datetime import datetime
import base64

def generate_image(prompt: str, output_path: str = None) -> dict:
    """テキストプロンプトから画像を生成"""
    client = genai.Client(api_key=os.getenv("GEMINI_API_KEY"))

    # 画像生成リクエスト
    response = client.models.generate_images(
        model="gemini-2.0-flash",
        prompt=prompt,
        number_of_images=1,
        size_width=1024,
        size_height=1024,
        safety_filter_level="block_none",
        person_generation_config=genai.types.PersonGenerationConfig(
            allow_adult=False
        )
    )

    # 画像データを取得
    if response.images:
        image = response.images[0]
        image_data = image.data

        # 画像をファイルに保存
        if output_path:
            with open(output_path, "wb") as f:
                f.write(image_data)
            print(f"画像を保存しました: {output_path}")
        else:
            # ファイル名を自動生成
            timestamp = datetime.now().strftime("%Y%m%d_%H%M%S")
            output_path = f"generated_image_{timestamp}.png"
            with open(output_path, "wb") as f:
                f.write(image_data)

        return {
            "status": "success",
            "path": output_path,
            "size": f"{len(image_data)} bytes"
        }
    else:
        return {
            "status": "failed",
            "error": "画像生成に失敗しました"
        }

# 使用例
prompt = "美しい夕焼けの風景、富士山のシルエット、湖面の反射"
result = generate_image(prompt, "sunset.png")

print(f"生成ステータス: {result['status']}")
print(f"ファイルパス: {result['path']}")

# 出力例:
# 生成ステータス: success
# ファイルパス: sunset.png
```

## プロンプトエンジニアリング

### 効果的なプロンプト設計

画像生成の品質は、プロンプトの質に大きく左右されます。具体的で詳細な指示を与えることが重要です。

```python
def create_detailed_prompt(
    subject: str,
    style: str,
    setting: str,
    mood: str,
    technical_details: str = None
) -> str:
    """詳細なプロンプトを構築"""
    prompt = f"""
    主体: {subject}
    スタイル: {style}
    設定: {setting}
    雰囲気: {mood}
    """

    if technical_details:
        prompt += f"\n技術的指定: {technical_details}"

    return prompt.strip()

def batch_generate_images(prompt_configs: list) -> list:
    """複数の画像設定で一括生成"""
    client = genai.Client(api_key=os.getenv("GEMINI_API_KEY"))
    results = []

    for i, config in enumerate(prompt_configs):
        # 詳細なプロンプトを作成
        prompt = create_detailed_prompt(
            subject=config.get("subject"),
            style=config.get("style"),
            setting=config.get("setting"),
            mood=config.get("mood"),
            technical_details=config.get("technical_details")
        )

        try:
            response = client.models.generate_images(
                model="gemini-2.0-flash",
                prompt=prompt,
                number_of_images=1,
                size_width=1024,
                size_height=1024
            )

            if response.images:
                filename = f"generated_{i:03d}.png"
                with open(filename, "wb") as f:
                    f.write(response.images[0].data)

                results.append({
                    "config_index": i,
                    "filename": filename,
                    "prompt": prompt,
                    "status": "success"
                })
        except Exception as e:
            results.append({
                "config_index": i,
                "error": str(e),
                "status": "failed"
            })

    return results

# 使用例
prompt_configs = [
    {
        "subject": "ロボット",
        "style": "サイバーパンク",
        "setting": "ネオンが輝く都市夜景",
        "mood": "未来的で神秘的",
        "technical_details": "高解像度、4K品質、メタリック素材"
    },
    {
        "subject": "古城",
        "style": "ファンタジーイラスト",
        "setting": "霧の中の山頂",
        "mood": "荘厳で物悲しい",
        "technical_details": "油絵風、暖色系"
    }
]

results = batch_generate_images(prompt_configs)
for result in results:
    if result["status"] == "success":
        print(f"✓ 生成成功: {result['filename']}")
    else:
        print(f"✗ 生成失敗: {result['error']}")

# 出力例:
# ✓ 生成成功: generated_000.png
# ✓ 生成成功: generated_001.png
```

## 画像のカスタマイズと最適化

### 様々なアスペクト比と解像度

```python
def generate_image_custom_size(
    prompt: str,
    width: int = 1024,
    height: int = 1024,
    num_images: int = 1
) -> dict:
    """カスタムサイズで画像を生成"""
    client = genai.Client(api_key=os.getenv("GEMINI_API_KEY"))

    # サポートされている解像度を確認
    supported_sizes = [
        (1024, 1024),
        (768, 1024),
        (1024, 768),
        (512, 512),
        (1344, 768)
    ]

    if (width, height) not in supported_sizes:
        print(f"警告: {width}x{height}はサポートされていません")
        width, height = 1024, 1024

    response = client.models.generate_images(
        model="gemini-2.0-flash",
        prompt=prompt,
        number_of_images=num_images,
        size_width=width,
        size_height=height
    )

    images = []
    for i, image in enumerate(response.images):
        filename = f"custom_{width}x{height}_{i}.png"
        with open(filename, "wb") as f:
            f.write(image.data)
        images.append(filename)

    return {
        "prompt": prompt,
        "resolution": f"{width}x{height}",
        "count": len(images),
        "files": images
    }

# 使用例
result = generate_image_custom_size(
    prompt="広い草原にいるシマウマの群れ",
    width=1024,
    height=768,
    num_images=3
)

print(f"解像度: {result['resolution']}")
print(f"生成画像数: {result['count']}")

# 出力例:
# 解像度: 1024x768
# 生成画像数: 3
```

## ベストプラクティスと注意点

### セーフティフィルタリング

```python
def generate_with_safety_controls(
    prompt: str,
    safety_level: str = "block_medium_and_above"
) -> dict:
    """セーフティフィルタを適用して画像生成"""
    client = genai.Client(api_key=os.getenv("GEMINI_API_KEY"))

    # セーフティレベルの選択肢
    safety_levels = {
        "block_none": "フィルタなし（カスタム用途）",
        "block_low_and_above": "低レベル以上をブロック",
        "block_medium_and_above": "中レベル以上をブロック",
        "block_high_and_above": "高レベル以上をブロック"
    }

    response = client.models.generate_images(
        model="gemini-2.0-flash",
        prompt=prompt,
        number_of_images=1,
        safety_filter_level=safety_level,
        size_width=1024,
        size_height=1024
    )

    return {
        "safety_level": safety_levels.get(safety_level),
        "generated": bool(response.images),
        "image_count": len(response.images) if response.images else 0
    }

# 使用例
result = generate_with_safety_controls(
    prompt="美しい風景画",
    safety_level="block_medium_and_above"
)

print(f"セーフティレベル: {result['safety_level']}")
print(f"生成成功: {result['generated']}")
```

## よくある質問

### Q1: 生成される画像の著作権は誰のものですか？

**A:** Gemini APIで生成した画像の著作権はあなた（ユーザー）に帰属します。商用利用も可能ですが、利用規約を確認することをお勧めします。ただし、生成された画像が既存の著作物に類似していないことを保証する責任はユーザーにあります。

### Q2: 実在の人物を生成することはできますか？

**A:** いいえ。セーフティフィルタにより、実在の著名人の顔を生成することはできません。ただし、フィクション的な人物キャラクターは生成可能です。

### Q3: 生成の失敗やリトライについてはどうしますか？

**A:** API呼び出しが失敗した場合は、プロンプトを簡潔にするか、別の表現に変更してリトライしてください。場合によっては、セーフティフィルタによってブロックされている可能性があります。
