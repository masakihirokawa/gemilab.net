## 音声理解APIの概要

Gemini 音声理解APIは、音声ファイルをテキストに変換し、その内容を分析・理解するための強力なツールです。以下のような用途に対応します：

- ポッドキャストやウェビナーの文字起こし
- 会議やインタビューの記録と要約
- 音声コンテンツの検索とインデックス化
- 言語学習や音声コンテンツの品質分析
- カスタマーサービス通話の自動分析

このガイドでは、APIの基本的な使用方法から実践的な応用まで、段階的に解説します。

## シンプルな音声文字起こし

### 基本的な音声ファイルの処理

```python
import mimetypes

def transcribe_audio(audio_file_path: str) -> dict:
    """音声ファイルを文字起こし"""
    client = genai.Client(api_key=os.getenv("GEMINI_API_KEY"))

    # ファイルの存在確認とMIMEタイプの決定
    if not Path(audio_file_path).exists():
        return {"status": "error", "message": "ファイルが見つかりません"}

    mime_type, _ = mimetypes.guess_type(audio_file_path)
    if not mime_type or 'audio' not in mime_type:
        # 一般的な音声フォーマットの場合
        file_ext = Path(audio_file_path).suffix.lower()
        mime_map = {
            '.mp3': 'audio/mpeg',
            '.wav': 'audio/wav',
            '.m4a': 'audio/mp4',
            '.flac': 'audio/flac',
            '.ogg': 'audio/ogg',
            '.webm': 'audio/webm'
        }
        mime_type = mime_map.get(file_ext, 'audio/mpeg')

    # 音声ファイルをアップロード
    with open(audio_file_path, 'rb') as f:
        audio_data = f.read()

    try:
        response = client.models.generate_content(
            model="gemini-2.0-flash",
            contents=[
                "この音声の内容を完全に文字起こししてください。",
                {
                    "mime_type": mime_type,
                    "data": audio_data
                }
            ]
        )

        transcript = response.text
        return {
            "status": "success",
            "file": audio_file_path,
            "transcript": transcript,
            "length": len(transcript)
        }

    except Exception as e:
        return {
            "status": "error",
            "message": str(e)
        }

# 使用例
result = transcribe_audio("meeting_recording.mp3")

if result["status"] == "success":
    print(f"文字起こし完了: {result['length']}文字")
    print("内容:")
    print(result["transcript"][:300] + "...")

# 出力例:
# 文字起こし完了: 4523文字
# 内容:
# 本日の会議にようこそ。今回のアジェンダは3つあります...
```

## 音声コンテンツの分析

### ポッドキャストの要約と分類

```python
def analyze_podcast(audio_file_path: str, analysis_type: str = "summary") -> dict:
    """ポッドキャストを分析・要約"""
    client = genai.Client(api_key=os.getenv("GEMINI_API_KEY"))

    # 分析タイプに応じたプロンプト
    analysis_prompts = {
        "summary": "この音声の要約を作成してください。最も重要なポイントを3-5個箇条書きで示してください。",
        "topics": "この音声で議論されている主なトピックをリストアップしてください。",
        "sentiment": "この音声全体の雰囲気と、スピーカーの感情トーンを分析してください。",
        "keywords": "この音声の重要なキーワードと専門用語を抽出してください。",
        "detailed": "この音声の詳細な説明を以下の構成で提供してください：\n1. 概要\n2. 主なテーマ\n3. 重要なポイント\n4. 結論・次のステップ"
    }

    prompt = analysis_prompts.get(analysis_type, analysis_prompts["summary"])

    # ファイル読み込みとMIMEタイプ決定
    mime_type = 'audio/mpeg'  # デフォルト
    if audio_file_path.endswith('.wav'):
        mime_type = 'audio/wav'
    elif audio_file_path.endswith('.m4a'):
        mime_type = 'audio/mp4'

    with open(audio_file_path, 'rb') as f:
        audio_data = f.read()

    try:
        response = client.models.generate_content(
            model="gemini-2.0-flash",
            contents=[
                prompt,
                {
                    "mime_type": mime_type,
                    "data": audio_data
                }
            ]
        )

        return {
            "status": "success",
            "analysis_type": analysis_type,
            "file": audio_file_path,
            "result": response.text
        }

    except Exception as e:
        return {
            "status": "error",
            "message": str(e)
        }

# 使用例
podcast_file = "episode_001.mp3"

# 要約の生成
summary = analyze_podcast(podcast_file, analysis_type="summary")
if summary["status"] == "success":
    print("ポッドキャスト要約:")
    print(summary["result"])

# トピックの抽出
topics = analyze_podcast(podcast_file, analysis_type="topics")
print("\n主なトピック:")
print(topics["result"])

# 出力例:
# ポッドキャスト要約:
# • AIと機械学習の最新動向について解説
# • 企業でのAI導入の課題と解決策
# • 今後5年間の技術予測
# • 聴者からの質問への回答
```

## バッチ処理と効率化

### 複数の音声ファイルを一括処理

```python
def batch_transcribe_audio(audio_files: list) -> list:
    """複数の音声ファイルを一括文字起こし"""
    client = genai.Client(api_key=os.getenv("GEMINI_API_KEY"))
    results = []

    for i, audio_file in enumerate(audio_files):
        print(f"[{i+1}/{len(audio_files)}] 処理中: {audio_file}")

        if not Path(audio_file).exists():
            results.append({
                "file": audio_file,
                "status": "error",
                "message": "ファイルが見つかりません"
            })
            continue

        # MIME タイプの決定
        mime_type = 'audio/mpeg'
        if audio_file.endswith('.wav'):
            mime_type = 'audio/wav'
        elif audio_file.endswith('.m4a'):
            mime_type = 'audio/mp4'
        elif audio_file.endswith('.flac'):
            mime_type = 'audio/flac'

        try:
            with open(audio_file, 'rb') as f:
                audio_data = f.read()

            response = client.models.generate_content(
                model="gemini-2.0-flash",
                contents=[
                    "この音声を文字起こししてください:",
                    {
                        "mime_type": mime_type,
                        "data": audio_data
                    }
                ]
            )

            transcript = response.text

            # 文字起こし結果をファイルに保存
            output_file = f"{Path(audio_file).stem}_transcript.txt"
            with open(output_file, 'w', encoding='utf-8') as f:
                f.write(transcript)

            results.append({
                "file": audio_file,
                "status": "success",
                "output": output_file,
                "transcript_length": len(transcript)
            })
            print(f"✓ 完了: {output_file}")

        except Exception as e:
            results.append({
                "file": audio_file,
                "status": "error",
                "message": str(e)
            })
            print(f"✗ エラー: {str(e)}")

    return results

# 使用例
audio_files = [
    "meeting_001.mp3",
    "meeting_002.mp3",
    "meeting_003.mp3"
]

batch_results = batch_transcribe_audio(audio_files)

# 結果のサマリー
successful = [r for r in batch_results if r["status"] == "success"]
print(f"\n処理完了: {len(successful)}/{len(audio_files)} 件成功")

# 出力例:
# [1/3] 処理中: meeting_001.mp3
# ✓ 完了: meeting_001_transcript.txt
# [2/3] 処理中: meeting_002.mp3
# ✓ 完了: meeting_002_transcript.txt
# [3/3] 処理中: meeting_003.mp3
# ✓ 完了: meeting_003_transcript.txt
#
# 処理完了: 3/3 件成功
```

## 高度な音声分析

### カスタム分析パイプライン

```python
def comprehensive_audio_analysis(audio_file: str) -> dict:
    """複合的な音声分析"""
    client = genai.Client(api_key=os.getenv("GEMINI_API_KEY"))

    with open(audio_file, 'rb') as f:
        audio_data = f.read()

    mime_type = 'audio/mpeg'

    # 複合分析プロンプト
    analysis_prompt = """
    この音声を以下の観点から詳細に分析してください：

    1. トランスクリプション（完全な文字起こし）
    2. スピーカー情報（スピーカー数、言語、イントネーション）
    3. 主要なテーマと議論のポイント
    4. アクションアイテム（必要に応じて）
    5. 感情分析（全体的な雰囲気、スピーカーの感情状態）
    6. キーワードと専門用語
    7. 推奨される次のステップ

    JSON形式で構造化して返してください。
    """

    try:
        response = client.models.generate_content(
            model="gemini-2.0-flash",
            contents=[
                analysis_prompt,
                {
                    "mime_type": mime_type,
                    "data": audio_data
                }
            ]
        )

        # 結果の保存
        output_file = f"{Path(audio_file).stem}_analysis.json"
        with open(output_file, 'w', encoding='utf-8') as f:
            f.write(response.text)

        return {
            "status": "success",
            "source": audio_file,
            "output_file": output_file,
            "analysis": response.text
        }

    except Exception as e:
        return {
            "status": "error",
            "message": str(e)
        }

# 使用例
analysis = comprehensive_audio_analysis("important_meeting.mp3")
if analysis["status"] == "success":
    print(f"分析結果を保存しました: {analysis['output_file']}")
```

## よくある質問

### Q1: どのような音声フォーマットをサポートしていますか？

**A:** API は MP3、WAV、M4A、FLAC、OGG、WebM などの一般的な音声フォーマットをサポートしています。ファイルのMIMEタイプを正しく指定することが重要です。

### Q2: 音声ファイルのサイズに制限はありますか？

**A:** はい。通常、1リクエストあたり最大25MBまでのファイルサイズをサポートしています。より大きなファイルは分割して処理するか、Batch APIを利用してください。

### Q3: 複数のスピーカーがいる場合、誰が何を話したかを識別できますか？

**A:** 基本的な文字起こしではスピーカー識別は自動的には行われません。ただし、プロンプトで明示的に依頼することで、スピーカーの区別を試みることは可能です。より高度な分析が必要な場合は、別の音声分析ツールの使用を検討してください。
