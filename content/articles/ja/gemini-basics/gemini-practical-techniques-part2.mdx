## 本番グレードのエージェントシステム

### Gemini 2.5 Pro で並列ツール呼び出し

複数のタスクを同時に実行することで、大幅なレイテンシ削減が可能です。Gemini 2.5 Pro はネイティブに並列ツール呼び出しに対応しています：

```python
import asyncio
import google.generativeai as genai
from concurrent.futures import ThreadPoolExecutor

# ツール定義：複数の外部 API 呼び出し
tools = [
    {
        "name": "get_weather",
        "description": "指定都市の天気情報を取得",
        "input_schema": {
            "type": "object",
            "properties": {
                "city": {"type": "string"}
            },
            "required": ["city"]
        }
    },
    {
        "name": "get_travel_cost",
        "description": "都市間の交通費を計算",
        "input_schema": {
            "type": "object",
            "properties": {
                "origin": {"type": "string"},
                "destination": {"type": "string"}
            },
            "required": ["origin", "destination"]
        }
    },
    {
        "name": "get_hotel_price",
        "description": "ホテルの平均価格を取得",
        "input_schema": {
            "type": "object",
            "properties": {
                "city": {"type": "string"},
                "stars": {"type": "integer", "enum": [3, 4, 5]}
            },
            "required": ["city", "stars"]
        }
    }
]

# 実装
def get_weather(city: str) -> dict:
    """実装の詳細は省略"""
    return {"city": city, "temp": 25, "condition": "sunny"}

def get_travel_cost(origin: str, destination: str) -> dict:
    return {"route": f"{origin}→{destination}", "cost": 12000}

def get_hotel_price(city: str, stars: int) -> dict:
    return {"city": city, "stars": stars, "avg_price": 15000 * stars}

# Model 初期化
model = genai.GenerativeModel("gemini-2.5-pro", tools=tools)

# ユーザーリクエスト
prompt = "東京から大阪への旅行計画を立ててください。天気、交通費、宿泊先（5 つ星）の情報が必要です。"

# 並列実行の処理
def execute_parallel_tools(tool_calls: list) -> list:
    """複数のツール呼び出しを並列実行"""
    results = []

    with ThreadPoolExecutor(max_workers=5) as executor:
        futures = {}

        for call in tool_calls:
            if call.name == "get_weather":
                future = executor.submit(get_weather, call.args["city"])
            elif call.name == "get_travel_cost":
                future = executor.submit(
                    get_travel_cost,
                    call.args["origin"],
                    call.args["destination"]
                )
            elif call.name == "get_hotel_price":
                future = executor.submit(
                    get_hotel_price,
                    call.args["city"],
                    call.args["stars"]
                )

            futures[call.name] = future

        # すべての結果を待つ
        for tool_name, future in futures.items():
            results.append({"tool": tool_name, "result": future.result()})

    return results

# AI の応答を処理
response = model.generate_content(prompt)

# 並列ツール呼び出しがある場合
parallel_results = []
for part in response.content:
    if hasattr(part, 'function_call'):
        # 複数の function_call が含まれている場合、すべてを抽出
        pass

# 完全な実装
print("旅行計画の作成準備完了...")
```

**重要：** Gemini 2.5 Pro は `function_call_list` を返し、複数のツール呼び出しを同時に指定します。アプリケーション側で並列実行を管理すれば、全体レイテンシは最も遅いツール呼び出しの時間のみになります。

**パフォーマンス比較：**
```
順序実行：get_weather (1s) + get_travel_cost (0.8s) + get_hotel_price (0.7s) = 2.5s
並列実行：max(1s, 0.8s, 0.7s) = 1s
```

### エラーハンドリングとリトライ戦略

本番環境ではネットワーク遅延、API 利用制限、タイムアウトが発生します：

```python
import time
from typing import Optional, Dict, Any
from tenacity import (
    retry,
    wait_exponential,
    stop_after_attempt,
    retry_if_exception_type,
)

class GeminiAPIError(Exception):
    pass

@retry(
    wait=wait_exponential(multiplier=1, min=2, max=10),  # 指数バックオフ
    stop=stop_after_attempt(3),  # 最大 3 回まで再試行
    retry=retry_if_exception_type((GeminiAPIError, Exception)),
    reraise=True
)
def call_gemini_with_retry(
    prompt: str,
    max_tokens: int = 1024
) -> Optional[str]:
    """リトライロジック付き API 呼び出し"""
    try:
        model = genai.GenerativeModel("gemini-2.5-pro")
        response = model.generate_content(
            prompt,
            generation_config=genai.types.GenerationConfig(
                max_output_tokens=max_tokens,
                temperature=0.7
            )
        )
        return response.text

    except genai.types.APIError as e:
        if e.status_code == 429:  # Rate limit
            raise GeminiAPIError(f"Rate limited: {e}") from e
        elif e.status_code == 500:  # Server error
            raise GeminiAPIError(f"Server error: {e}") from e
        else:
            raise

    except Exception as e:
        raise GeminiAPIError(f"Unexpected error: {e}") from e


# 使用例
try:
    result = call_gemini_with_retry("日本の GNP について説明してください")
    print(result)
except GeminiAPIError as e:
    print(f"3 回の再試行後も失敗：{e}")
```

**ベストプラクティス：**
1. 指数バックオフで API 負荷を分散
2. 429（Rate Limit）には長めの待機時間
3. 5xx エラーは再試行の対象
4. 4xx エラー（400、401、403）は再試行しない
5. タイムアウトは適切に設定（通常 30～60 秒）

## Live API × 音声 AI

### リアルタイム音声アシスタントの構築

Gemini 2.5 Pro の Live API により、WebSocket 経由でリアルタイム音声対話が可能です：

```python
import asyncio
import websockets
import json
import base64
import sounddevice as sd
import numpy as np

async def audio_assistant():
    """リアルタイム音声会話エージェント"""

    # WebSocket URI（実際の URI は Google AI API ドキュメント参照）
    ws_uri = "wss://generativelanguage.googleapis.com/google.ai.generativelanguage.v1alpha.GenerativeService/BidiGenerateContent"

    async with websockets.connect(
        ws_uri,
        subprotocols=["json", "proto"],
        extra_headers={"Authorization": f"Bearer {GEMINI_API_KEY}"}
    ) as websocket:

        # セッション初期化
        setup_msg = {
            "setup": {
                "model": "models/gemini-2.5-pro",
                "generation_config": {
                    "temperature": 0.7,
                    "max_output_tokens": 1024
                },
                "system_instruction": "You are a helpful Japanese-speaking AI assistant."
            }
        }
        await websocket.send(json.dumps(setup_msg))

        # 初期化応答を受け取る
        init_response = await websocket.recv()
        print(f"Init response: {init_response}")

        # 音声入力スレッド
        async def send_audio():
            """マイクからの入力を送信"""
            print("音声アシスタント開始。話しかけてください...")

            # 音声キャプチャ（16kHz、単チャンネル）
            sample_rate = 16000
            chunk_duration = 0.5  # 500ms ごとに送信
            chunk_samples = int(sample_rate * chunk_duration)

            with sd.InputStream(
                channels=1,
                samplerate=sample_rate,
                dtype=np.int16
            ) as stream:
                while True:
                    audio_chunk, _ = stream.read(chunk_samples)

                    # 音声をバイナリにエンコード
                    audio_bytes = audio_chunk.tobytes()
                    audio_b64 = base64.b64encode(audio_bytes).decode()

                    # Gemini に送信
                    msg = {
                        "realtime_input": {
                            "media_chunks": [
                                {
                                    "data": audio_b64,
                                    "mime_type": "audio/pcm"
                                }
                            ]
                        }
                    }
                    await websocket.send(json.dumps(msg))
                    await asyncio.sleep(chunk_duration)

        # 音声出力スレッド
        async def receive_audio():
            """Gemini からの応答を受け取る"""
            while True:
                try:
                    response = await asyncio.wait_for(websocket.recv(), timeout=1.0)
                    data = json.loads(response)

                    # テキスト応答
                    if "server_content" in data:
                        for part in data["server_content"]["parts"]:
                            if "text" in part:
                                print(f"AI: {part['text']}")

                    # 音声応答（TTS）
                    if "server_content" in data:
                        for part in data["server_content"]["parts"]:
                            if "inline_data" in part:
                                audio_b64 = part["inline_data"]["data"]
                                audio_bytes = base64.b64decode(audio_b64)
                                # スピーカー出力
                                sd.play(
                                    np.frombuffer(audio_bytes, np.int16),
                                    samplerate=16000
                                )

                except asyncio.TimeoutError:
                    continue
                except Exception as e:
                    print(f"Error: {e}")
                    break

        # 並行実行
        send_task = asyncio.create_task(send_audio())
        recv_task = asyncio.create_task(receive_audio())

        await asyncio.gather(send_task, recv_task)

# 実行
asyncio.run(audio_assistant())
```

### コンテキストキャッシュでコスト最適化

大規模な入力（PDF、長文など）を複数回利用する場合、キャッシュを活用してコスト削減：

```python
import google.generativeai as genai
from google.generativeai import caching
import hashlib

# 大規模なシステムプロンプト
system_prompt = """
You are an expert in Japanese business law.
Follow these guidelines:
1. Always cite relevant law articles
2. Provide practical advice
3. Acknowledge uncertainty
...（500～1000 行の詳細ガイドライン）
"""

# キャッシュポリシー設定（24 時間）
cache_config = caching.CachedContent(
    model="models/gemini-2.5-pro",
    system_instruction=system_prompt,
    ttl_seconds=86400,  # 24 時間
)

# キャッシュを作成
cached_content = genai.caching.cached_content.create(cache_config)
cache_id = cached_content.name
print(f"キャッシュ作成：{cache_id}")

# キャッシュ付きでリクエスト（大幅にコスト削減）
model = genai.GenerativeModel(
    "gemini-2.5-pro",
    cached_content=cache_id
)

# 複数のユーザーリクエストに対応
queries = [
    "契約書の無効理由について",
    "取締役の責任について",
    "労働法の最新改正について"
]

for query in queries:
    response = model.generate_content(query)
    print(f"Q: {query}\nA: {response.text}\n---")

# コスト比較：
# キャッシュなし：1000 トークン × 3 リクエスト = 3000 トークン課金
# キャッシュあり：キャッシュセットアップ（1000） + 20 トークン × 3 = 1060 トークン（約 65% 削減）
```

### Veo 3 動画生成 API

テキストから動画を自動生成（Gemini 統合）：

```python
from google.cloud import generativelanguage_v1
import os

def generate_video_from_text(prompt: str, duration_seconds: int = 6) -> str:
    """テキストプロンプトから動画を生成"""

    client = generativelanguage_v1.GenerativeServiceClient()

    request = generativelanguage_v1.GenerateContentRequest(
        model="models/veo-3",  # 動画生成モデル
        contents=[
            generativelanguage_v1.Content(
                parts=[
                    generativelanguage_v1.Part(
                        text=prompt
                    )
                ]
            )
        ],
        generation_config=generativelanguage_v1.GenerationConfig(
            temperature=0.8,
            max_output_tokens=2048,
            # Video 固有のパラメータ
            candidate_count=1,  # 生成する動画数
        )
    )

    response = client.generate_content(request)

    # 動画 URL を取得
    video_url = response.candidates[0].content.parts[0].file_data.uri
    return video_url

# 使用例
prompt = """
春の京都、清水寺の参道を歩く観光客の風景。
桜の花びらが舞う中、門をくぐり、本堂へ向かう。
ゆっくりしたカメラワーク。日本の伝統的な美しさを表現。
"""

video_url = generate_video_from_text(prompt, duration_seconds=8)
print(f"生成された動画：{video_url}")

# 動画をダウンロード・編集・公開可能
```

### Gemini API × Stripe で AI サービスを収益化

Gemini API を使った SaaS の収益化モデル：

```python
import stripe
import google.generativeai as genai
from functools import wraps
from datetime import datetime, timedelta

# Stripe 設定
stripe.api_key = "sk_live_xxx"

# API 利用量の追跡
class APIUsageTracker:
    def __init__(self):
        self.usage_log = {}  # {user_id: {date: tokens_used}}

    def track_usage(self, user_id: str, tokens: int):
        """ユーザーの API 利用を記録"""
        today = str(datetime.now().date())
        if user_id not in self.usage_log:
            self.usage_log[user_id] = {}
        if today not in self.usage_log[user_id]:
            self.usage_log[user_id][today] = 0
        self.usage_log[user_id][today] += tokens

    def get_monthly_usage(self, user_id: str) -> int:
        """月間トークン使用量を取得"""
        total = 0
        today = datetime.now().date()
        month_start = today.replace(day=1)
        for date_str, tokens in self.usage_log.get(user_id, {}).items():
            date = datetime.strptime(date_str, "%Y-%m-%d").date()
            if date >= month_start:
                total += tokens
        return total

tracker = APIUsageTracker()

# 利用量ベースの課金関数
def charge_api_usage(user_id: str, tokens_used: int):
    """API 利用量に基づき Stripe で課金"""
    # 価格設定：1M トークン = $20
    cost = (tokens_used / 1_000_000) * 20

    # ユーザーの Stripe Customer を取得
    customer_id = get_customer_id(user_id)  # 実装省略

    # 課金実行
    if cost > 0.01:  # 最小課金額 $0.01
        charge = stripe.Charge.create(
            amount=int(cost * 100),  # セント単位
            currency="usd",
            customer=customer_id,
            description=f"Gemini API usage - {tokens_used} tokens"
        )
        print(f"Charged {user_id} for {tokens_used} tokens: ${cost}")

    # 利用量を追跡
    tracker.track_usage(user_id, tokens_used)

# デコレータで自動課金
def requires_payment(func):
    @wraps(func)
    def wrapper(user_id: str, *args, **kwargs):
        # 月間制限チェック（例：100M トークン以上で追加料金）
        monthly_usage = tracker.get_monthly_usage(user_id)
        if monthly_usage > 100_000_000:
            # Premium プランへのアップグレード提案
            print(f"Warning: {user_id} exceeded free tier. Charging for overage.")

        result = func(user_id, *args, **kwargs)

        # 使用トークン数を取得して課金
        if hasattr(result, "usage_metadata"):
            charge_api_usage(user_id, result.usage_metadata.output_tokens)

        return result

    return wrapper

@requires_payment
def generate_content_with_billing(user_id: str, prompt: str) -> str:
    """課金対象の API 呼び出し"""
    model = genai.GenerativeModel("gemini-2.5-pro")
    response = model.generate_content(prompt)
    # トークン情報を付加
    response.usage_metadata = response._raw.usage_metadata
    return response

# 使用例
result = generate_content_with_billing(
    user_id="user_123",
    prompt="Python の非同期プログラミングについて説明してください"
)
print(result.text)
```
