## コアモジュール 1: メモリ管理 (`memory.py`)

メモリシステムは基盤です。これがないと、AIパートナーは毎回新しく始まります。

```python
import json
import os
from datetime import datetime
from typing import Dict, List, Optional

class MemoryManager:
    def __init__(self, companion_name: str, data_dir: str = "./companion_data"):
        self.companion_name = companion_name
        self.data_dir = data_dir
        self.memory_file = os.path.join(data_dir, f"{companion_name}_memory.json")
        self.conversation_file = os.path.join(data_dir, f"{companion_name}_conversations.json")

        os.makedirs(data_dir, exist_ok=True)
        self._initialize_memory()

    def _initialize_memory(self):
        """メモリファイルが存在しない場合は作成します。"""
        if not os.path.exists(self.memory_file):
            initial_data = {
                "companion_name": self.companion_name,
                "created": datetime.now().isoformat(),
                "user_profile": {},
                "conversation_history": [],
                "key_insights": [],
                "preferences": {}
            }
            self._save_memory(initial_data)

        if not os.path.exists(self.conversation_file):
            with open(self.conversation_file, 'w', encoding='utf-8') as f:
                json.dump([], f, ensure_ascii=False)

    def add_to_history(self, role: str, content: str):
        """会話履歴にメッセージを追加。"""
        data = self._load_memory()
        data["conversation_history"].append({
            "timestamp": datetime.now().isoformat(),
            "role": role,
            "content": content
        })
        # 最後の50メッセージを保持してコンテキストウィンドウを管理
        data["conversation_history"] = data["conversation_history"][-50:]
        self._save_memory(data)

    def get_context_window(self, limit: int = 20) -> str:
        """フォーマットされた最近の会話コンテキストを返す。"""
        data = self._load_memory()
        recent = data["conversation_history"][-limit:]

        context = "最近の会話履歴:\n"
        for msg in recent:
            role = "ユーザー" if msg["role"] == "user" else "アシスタント"
            context += f"{role}: {msg['content']}\n"

        return context

    def update_user_profile(self, updates: Dict):
        """ユーザーについて保存されている情報を更新。"""
        data = self._load_memory()
        data["user_profile"].update(updates)
        self._save_memory(data)

    def add_insight(self, insight: str):
        """ユーザーについての重要な気づきを保存。"""
        data = self._load_memory()
        data["key_insights"].append({
            "timestamp": datetime.now().isoformat(),
            "insight": insight
        })
        self._save_memory(data)

    def _load_memory(self) -> Dict:
        with open(self.memory_file, 'r', encoding='utf-8') as f:
            return json.load(f)

    def _save_memory(self, data: Dict):
        with open(self.memory_file, 'w', encoding='utf-8') as f:
            json.dump(data, f, indent=2, ensure_ascii=False)

    def get_full_context(self) -> str:
        """完全なメモリコンテキストを取得してプロンプトに注入。"""
        data = self._load_memory()

        context = "=== ユーザープロフィール ===\n"
        for key, value in data["user_profile"].items():
            context += f"{key}: {value}\n"

        context += "\n=== 重要な洞察 ===\n"
        for insight_obj in data["key_insights"][-5:]:
            context += f"- {insight_obj['insight']}\n"

        context += "\n=== 最近の履歴 ===\n"
        context += self.get_context_window(10)

        return context
```

## コアモジュール 2: ペルソナ管理 (`persona.py`)

会話コンテキストとムードに基づいて適応する動的ペルソナ。

```python
import json
import os
from typing import Dict, Optional

class PersonaManager:
    def __init__(self, personas_dir: str = "./personas"):
        self.personas_dir = personas_dir
        os.makedirs(personas_dir, exist_ok=True)

    def create_persona(self, name: str, persona_config: Dict):
        """新しいペルソナ設定を作成。"""
        persona_file = os.path.join(self.personas_dir, f"{name}.json")

        default_config = {
            "name": name,
            "base_instructions": "",
            "tone": "professional",
            "expertise": [],
            "boundaries": [],
            "response_style": "balanced",
            "moods": {}  # 適応的なムードベースの指示
        }

        default_config.update(persona_config)

        with open(persona_file, 'w', encoding='utf-8') as f:
            json.dump(default_config, f, indent=2, ensure_ascii=False)

    def load_persona(self, name: str) -> Dict:
        """ペルソナ設定を読み込む。"""
        persona_file = os.path.join(self.personas_dir, f"{name}.json")
        with open(persona_file, 'r', encoding='utf-8') as f:
            return json.load(f)

    def build_system_prompt(self, persona_name: str,
                           user_context: str,
                           current_mood: Optional[str] = None) -> str:
        """コンテキスト付きの完全なシステムプロンプトを構築。"""
        persona = self.load_persona(persona_name)

        prompt = f"あなたは{persona['name']}です。\n\n"
        prompt += f"基本指示:\n{persona['base_instructions']}\n\n"

        if current_mood and current_mood in persona.get("moods", {}):
            prompt += f"ユーザーの現在のムード: {current_mood}\n"
            prompt += f"適応: {persona['moods'][current_mood]}\n\n"

        prompt += f"トーン: {persona['tone']}\n"
        prompt += f"応答スタイル: {persona['response_style']}\n\n"

        if persona.get("boundaries"):
            prompt += "境界:\n"
            for boundary in persona["boundaries"]:
                prompt += f"- {boundary}\n"

        prompt += f"\n=== ユーザーコンテキスト ===\n{user_context}"

        return prompt

    def list_personas(self) -> list:
        """利用可能なすべてのペルソナをリスト。"""
        return [f.replace('.json', '')
                for f in os.listdir(self.personas_dir)
                if f.endswith('.json')]
```

## コアモジュール 3: 感情トラッキング (`context.py`)

センチメント分析と感情状態管理。

```python
from typing import Tuple
from datetime import datetime
import json

class EmotionTracker:
    def __init__(self, memory_manager):
        self.memory = memory_manager
        self.emotion_scale = {
            "very_negative": -2,
            "negative": -1,
            "neutral": 0,
            "positive": 1,
            "very_positive": 2
        }

    def analyze_sentiment(self, text: str) -> Tuple[str, float]:
        """
        テキストからユーザーのセンチメントを分析。
        本番環境では、適切なセンチメント分析ライブラリを使用。
        """
        # 簡略化されたセンチメント分析
        negative_words = ["悲しい", "怒っている", "落ち込んでいる", "嫌い", "大変"]
        positive_words = ["幸せ", "興奮している", "素晴らしい", "好き", "最高"]

        text_lower = text.lower()
        negative_count = sum(1 for word in negative_words if word in text_lower)
        positive_count = sum(1 for word in positive_words if word in text_lower)

        if negative_count > positive_count:
            if negative_count > 2:
                return "very_negative", self.emotion_scale["very_negative"]
            return "negative", self.emotion_scale["negative"]
        elif positive_count > negative_count:
            if positive_count > 2:
                return "very_positive", self.emotion_scale["very_positive"]
            return "positive", self.emotion_scale["positive"]

        return "neutral", self.emotion_scale["neutral"]

    def track_emotion(self, user_message: str) -> str:
        """感情データを追跡して保存。"""
        mood, score = self.analyze_sentiment(user_message)

        data = self.memory._load_memory()
        if "emotion_history" not in data:
            data["emotion_history"] = []

        data["emotion_history"].append({
            "timestamp": datetime.now().isoformat(),
            "mood": mood,
            "score": score
        })

        self.memory._save_memory(data)
        return mood

    def get_recent_mood(self) -> str:
        """ユーザーの最近の感情的傾向を取得。"""
        data = self.memory._load_memory()
        emotions = data.get("emotion_history", [])[-10:]

        if not emotions:
            return "neutral"

        avg_score = sum(e["score"] for e in emotions) / len(emotions)

        if avg_score < -1:
            return "very_negative"
        elif avg_score < 0:
            return "negative"
        elif avg_score > 1:
            return "very_positive"
        elif avg_score > 0:
            return "positive"

        return "neutral"
```

## コアモジュール 4: メインアプリケーション (`main.py`)

Gemini API 統合とすべてをまとめる。

```python
import os
import dotenv
import google.generativeai as genai
from memory import MemoryManager
from persona import PersonaManager
from context import EmotionTracker

dotenv.load_dotenv()
genai.configure(api_key=os.getenv("GEMINI_API_KEY"))

class GeminiCompanion:
    def __init__(self, companion_name: str, persona_name: str):
        self.companion_name = companion_name
        self.persona_name = persona_name
        self.memory = MemoryManager(companion_name)
        self.personas = PersonaManager()
        self.emotions = EmotionTracker(self.memory)
        self.model = genai.GenerativeModel("gemini-pro")

    def chat(self, user_message: str) -> str:
        """ユーザーメッセージを処理して応答を生成。"""

        # 感情を追跡
        mood = self.emotions.track_emotion(user_message)

        # コンテキストを使用してシステムプロンプトを構築
        user_context = self.memory.get_full_context()
        system_prompt = self.personas.build_system_prompt(
            self.persona_name,
            user_context,
            mood
        )

        # 履歴に追加
        self.memory.add_to_history("user", user_message)

        # 応答を生成
        full_prompt = f"{system_prompt}\n\nユーザーメッセージ: {user_message}"

        response = self.model.generate_content(full_prompt)
        assistant_message = response.text

        # 応答を保存
        self.memory.add_to_history("assistant", assistant_message)

        return assistant_message

    def start_interactive_session(self):
        """対話型会話ループを実行。"""
        print(f"\n🤖 {self.companion_name} ({self.persona_name}ペルソナを使用)")
        print("会話を終了するには「exit」と入力してください\n")

        while True:
            user_input = input("あなた: ").strip()

            if user_input.lower() == 'exit':
                print("さようなら！")
                break

            if not user_input:
                continue

            response = self.chat(user_input)
            print(f"\n{self.companion_name}: {response}\n")

    def update_user_knowledge(self, key: str, value: str):
        """コンパニオンがユーザーについて知っていることを更新。"""
        self.memory.update_user_profile({key: value})

# 使用例
if __name__ == "__main__":
    # ペルソナを作成
    personas_mgr = PersonaManager()

    personas_mgr.create_persona("thoughtful_advisor", {
        "base_instructions": """あなたは思慮深いキャリアアドバイザーです。
        あなたの役割: ユーザーがキャリア決定を慎重に考えるのを手伝う
        - アドバイスを提供する前に明確化の質問をする
        - 決定の感情的な側面を認める
        - 関連する場合はフレームワークを使用
        - 直接的だが親切に""",
        "tone": "温かいが専門的",
        "expertise": ["キャリア転換", "技術キャリア", "リーダーシップ"],
        "boundaries": ["財務アドバイザーではない", "医療・法律の質問をリダイレクト"],
        "moods": {
            "very_negative": "さらに支援的。問題よりも可能性に焦点を当てる。",
            "negative": "懸念を認めてから解決策にピボット。",
            "positive": "成功を祝い、勢いを構築。"
        }
    })

    # コンパニオンを作成して実行
    companion = GeminiCompanion("Alex", "thoughtful_advisor")

    # ユーザーについてコンパニオンに教える
    companion.update_user_knowledge("name", "Jordan")
    companion.update_user_knowledge("role", "シニアソフトウェアエンジニア")
    companion.update_user_knowledge("considering", "プロダクトマネジメントへの転職")

    # 会話を開始
    companion.start_interactive_session()
```
