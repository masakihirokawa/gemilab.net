## Stitch × Gemini × Antigravity — AIデザインから実装までの統合開発パイプライン

2026 年時点で、AIが支援する開発パイプラインは、デザイン、アーキテクチャ設計、実装の各段階で個別最適化されています。しかし最大の効果を生み出すのは、これら3つの段階を**統合パイプライン**として機能させることです。

本記事では、Google Stitch でデザイン → Gemini でアーキテクチャ設計 → Antigravity で実装、という 3 段階の完全統合ワークフローを構築する方法を、実践的なプロジェクト例を通じて解説します。

// ============ Business Logic ============

function calculateROI(property: Property, purchasePrice: number, holdingYears: number): number {
  // ROI = (Ending Value + Total Income - Beginning Value) / Beginning Value

  const appreciation = property.price * Math.pow(1 + property.appreciation_rate, holdingYears);
  const rentalIncome = property.rental_yield * purchasePrice * holdingYears;

  const roi = (appreciation + rentalIncome - purchasePrice) / purchasePrice;
  return roi * 100;  // percentage
}

function filterProperties(
  properties: Property[],
  criteria: {
    minPrice?: number;
    maxPrice?: number;
    minROI?: number;
    location?: string;
  }
): Property[] {
  return properties.filter(p => {
    if (criteria.minPrice && p.price < criteria.minPrice) return false;
    if (criteria.maxPrice && p.price > criteria.maxPrice) return false;
    if (criteria.minROI && p.roi_estimate < criteria.minROI) return false;
    if (criteria.location && p.location.city !== criteria.location) return false;
    return true;
  });
}

function buildComparisonMatrix(properties: Property[]): object {
  const metrics = ['price', 'roi_estimate', 'cap_rate', 'rental_yield', 'appreciation_rate'];

  return properties.reduce((acc, prop) => {
    acc[prop.id] = {};
    metrics.forEach(metric => {
      acc[prop.id][metric] = prop[metric];
    });
    return acc;
  }, {});
}

// src/lib/api/db.ts
import { createClient } from '@supabase/supabase-js';

const supabase = createClient(
  process.env.NEXT_PUBLIC_SUPABASE_URL!,
  process.env.SUPABASE_SERVICE_ROLE_KEY!
);

export const propertyDB = {
  async getProperties(filters?: {
    minPrice?: number;
    maxPrice?: number;
    location?: string;
    minROI?: number;
    sortBy?: 'price' | 'roi' | 'rating';
    limit?: number;
    offset?: number;
  }) {
    let query = supabase.from('properties').select('*');

    if (filters?.minPrice) query = query.gte('price', filters.minPrice);
    if (filters?.maxPrice) query = query.lte('price', filters.maxPrice);
    if (filters?.location) query = query.eq('city', filters.location);
    if (filters?.minROI) query = query.gte('roi_estimate', filters.minROI);

    if (filters?.sortBy === 'price') query = query.order('price', { ascending: true });
    if (filters?.sortBy === 'roi') query = query.order('roi_estimate', { ascending: false });

    const limit = Math.min(filters?.limit ?? 20, 100);
    const offset = filters?.offset ?? 0;

    query = query.range(offset, offset + limit - 1);

    const { data, error, count } = await query;
    if (error) throw error;

    return { properties: data, total: count };
  },

  async getPropertyById(id: string) {
    const { data, error } = await supabase
      .from('properties')
      .select('*')
      .eq('id', id)
      .single();

    if (error) throw error;
    return data;
  },

  async calculateROI(propertyId: string, purchasePrice: number, holdingYears: number) {
    const property = await this.getPropertyById(propertyId);

    const appreciation = property.price * Math.pow(1 + property.appreciation_rate, holdingYears);
    const rentalIncome = property.rental_yield * purchasePrice * holdingYears;

    const roi = (appreciation + rentalIncome - purchasePrice) / purchasePrice;

    return {
      purchase_price: purchasePrice,
      estimated_appreciation: appreciation - property.price,
      estimated_rental_income: rentalIncome,
      total_return: appreciation + rentalIncome - purchasePrice,
      roi_percentage: roi * 100,
    };
  },
};

// src/lib/api/routes.ts
import { NextRequest, NextResponse } from 'next/server';
import { propertyDB } from './db';

export async function GET(req: NextRequest) {
  const { searchParams } = new URL(req.url);

  const filters = {
    minPrice: searchParams.get('min_price') ? parseInt(searchParams.get('min_price')!) : undefined,
    maxPrice: searchParams.get('max_price') ? parseInt(searchParams.get('max_price')!) : undefined,
    location: searchParams.get('location') || undefined,
    minROI: searchParams.get('min_roi') ? parseFloat(searchParams.get('min_roi')!) : undefined,
    sortBy: (searchParams.get('sort_by') as any) || 'price',
    limit: searchParams.get('limit') ? parseInt(searchParams.get('limit')!) : 20,
    offset: searchParams.get('offset') ? parseInt(searchParams.get('offset')!) : 0,
  };

  try {
    const result = await propertyDB.getProperties(filters);
    return NextResponse.json(result);
  } catch (error) {
    return NextResponse.json({ error: 'Internal Server Error' }, { status: 500 });
  }
}

### Stitch MCP Server との連携

Stitch デザイン → Gemini API → Antigravity のパイプラインを自動化するため、Stitch MCP サーバーを構築します。

```typescript
// stitch-mcp-server.ts
import { Server } from '@modelcontextprotocol/sdk/server/index.js';
import { StdioServerTransport } from '@modelcontextprotocol/sdk/server/stdio.js';

const server = new Server({
  name: 'stitch-mcp-server',
  version: '1.0.0',
});

server.setRequestHandler('resources/read', async (uri) => {
  if (uri.startsWith('stitch://')) {
    const projectId = uri.split('://')[1];

    // Stitch API から デザインスキーマ取得
    const designSchema = await fetchStitchDesign(projectId);

    return {
      contents: [{
        uri,
        mimeType: 'application/json',
        text: JSON.stringify(designSchema),
      }],
    };
  }
});

server.setRequestHandler('tools/call', async (request) => {
  if (request.params.name === 'export_to_gemini') {
    const { projectId, outputFormat } = request.params.arguments;

    // 1. Stitch からデザインをエクスポート
    const designSchema = await fetchStitchDesign(projectId);

    // 2. Gemini に送信して API スキーマ生成
    const apiSchema = await callGeminiAPI(
      'Design to API transformation',
      designSchema
    );

    // 3. Antigravity へ自動送信
    const antigravityProject = await sendToAntigravity(
      projectId,
      apiSchema
    );

    return {
      content: [{
        type: 'text',
        text: `Project exported successfully. Antigravity project ID: ${antigravityProject.id}`,
      }],
    };
  }
});

async function fetchStitchDesign(projectId: string) {
  const response = await fetch(`https://stitch.google.com/api/v1/projects/${projectId}/export`, {
    headers: {
      'Authorization': `Bearer ${process.env.STITCH_API_KEY}`,
    },
  });
  return response.json();
}

async function callGeminiAPI(task: string, data: any) {
  // Gemini API 呼び出し
  const response = await fetch('https://generativelanguage.googleapis.com/v1beta/models/gemini-2.0-pro:generateContent', {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json',
      'x-goog-api-key': process.env.GEMINI_API_KEY,
    },
    body: JSON.stringify({
      contents: [{
        parts: [{
          text: `${task}\n\nData: ${JSON.stringify(data)}`,
        }],
      }],
    }),
  });

  const result = await response.json();
  return result.candidates[0].content.parts[0].text;
}

async function sendToAntigravity(projectId: string, apiSchema: any) {
  const response = await fetch('https://api.antigravitylab.net/projects/create', {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json',
      'Authorization': `Bearer ${process.env.ANTIGRAVITY_API_KEY}`,
    },
    body: JSON.stringify({
      name: `RE_${projectId}`,
      template: 'next.js',
      api_schema: apiSchema,
    }),
  });

  return response.json();
}

// サーバーを起動
const transport = new StdioServerTransport();
server.connect(transport).catch(console.error);
```
